\section{Laman's theorem}
\label{sec:laman-theorem}

For $n \ge 2$ vertices, a simple graph is generically rigid in the
plane iff it contains a Laman spanning subgraph (a $(2, 3)$-tight
subgraph on the same vertex set). The two directions land separately:

\begin{itemize}
  \item \emph{$\Leftarrow$ (every Laman graph is generically rigid in dim 2).}
    A Henneberg induction on $|V|$. Base case $|V| = 2$: $K_2$ is
    generically rigid (\cref{thm:top-fin-two-isGenericallyRigidInj}).
    Inductive step: peel off a Type~I or Type~II move via
    \cref{thm:isLaman-exists-typeI-or-typeII-reverse} and apply the
    move-rigidity-preservation theorem
    (\cref{thm:typeI-isGenericallyRigidInj-two} or
    \cref{thm:typeII-isGenericallyRigidInj-two}). The induction
    maintains the strong predicate \emph{injectively} generically rigid
    (\cref{def:isGenericallyRigidInj}), then weakens at the end.
  \item \emph{$\Rightarrow$ (every generically rigid graph contains a
    Laman spanning subgraph).} The Lov\'asz--Yemini~\cite{lovaszYemini1982}
    direction, which identifies the rigidity matroid in dim 2 with the
    $(2, 3)$-count matroid. This is Phase~6 and not yet formalized.
\end{itemize}

\subsection{The \texorpdfstring{$d = 2$}{d=2} edge-count corollary}

\begin{lemma}[Edge-count bound in dim 2]
  \label{lem:isGenericallyRigid-card-mul-le-two}
  \lean{SimpleGraph.IsGenericallyRigid.card_mul_le_two}
  \leanok
  \uses{thm:isGenericallyRigid-card-mul-le}
  Let $V$ be finite. If $G$ is generically rigid in dimension $2$, then
  \[
    2 |V| \;\le\; |E(G)| + 3.
  \]
\end{lemma}
\begin{proof}
  \leanok
  Specialize \cref{thm:isGenericallyRigid-card-mul-le} at $d = 2$.
\end{proof}

\subsection{The \texorpdfstring{$\Leftarrow$}{<=} direction}

\begin{theorem}[Every Laman graph is generically rigid in dim 2]
  \label{thm:isLaman-isGenericallyRigid-two}
  \lean{SimpleGraph.IsLaman.isGenericallyRigid_two}
  \leanok
  \uses{def:isLaman, def:isGenericallyRigid, def:isGenericallyRigidInj,
    thm:isLaman-exists-typeI-or-typeII-reverse,
    thm:typeI-isGenericallyRigidInj-two,
    thm:typeII-isGenericallyRigidInj-two,
    thm:top-fin-two-isGenericallyRigidInj,
    lem:isLaman-eq-top-of-card-eq-two,
    lem:isGenericallyRigidInj-iso}
  If $G$ is a Laman graph on a finite vertex type $V$, then $G$ is
  generically rigid in dimension $2$.
\end{theorem}
\begin{proof}
  \leanok
  Strong induction on $|V|$, proving the stronger statement: every
  Laman graph is injectively generically rigid in dimension $2$
  (\cref{def:isGenericallyRigidInj}); the actual theorem then weakens
  to plain generic rigidity at the end.

  \emph{Base case ($|V| \le 2$).} Tightness $|E(G)| + 3 = 2 |V|$ is
  infeasible in $\N$ for $|V| \le 1$, so $|V| = 2$ and by
  \cref{lem:isLaman-eq-top-of-card-eq-two} we have $G = K_2$ (on $V$).
  Transport \cref{thm:top-fin-two-isGenericallyRigidInj} at dimension
  $2$ along the bijection $V \simeq \mathrm{Fin}\,2$ using
  \cref{lem:isGenericallyRigidInj-iso}.

  \emph{Inductive step ($|V| \ge 3$).} By
  \cref{thm:isLaman-exists-typeI-or-typeII-reverse} pick $v$ and a
  Laman $G'$ on $\{w \ne v\}$ such that
  $G \simeq \mathrm{typeI}\,G'\,a\,b$ or
  $G \simeq \mathrm{typeII}\,G'\,a\,b\,c$. The vertex type
  $\{w \ne v\}$ has cardinality $|V| - 1$, so the induction hypothesis
  applies to $G'$: $G'$ is injectively generically rigid in dim $2$.
  Applying the appropriate move-preservation theorem
  (\cref{thm:typeI-isGenericallyRigidInj-two} or
  \cref{thm:typeII-isGenericallyRigidInj-two}) gives the same for
  $\mathrm{typeI}\,G'\,a\,b$ or $\mathrm{typeII}\,G'\,a\,b\,c$, and
  iso transport (\cref{lem:isGenericallyRigidInj-iso}) carries the
  conclusion back to $G$.
\end{proof}

\subsection{The \texorpdfstring{$\Rightarrow$}{=>} direction}

The companion direction is the subject of Phase~6, currently in
progress. The skeleton below lays out the dependency graph: each
intermediate definition / lemma is recorded without
\texttt{\textbackslash lean} or \texttt{\textbackslash leanok}, so the
dep-graph view renders them red until the corresponding Lean
declarations land.

\medskip

The statement we are proving is the necessary direction of Laman's
original theorem~\cite{laman1970}: every graph generically rigid in
the plane contains a Laman spanning subgraph. We work in the
linear-algebraic formulation of infinitesimal rigidity due to
Asimow--Roth~\cite{asimowRoth1978}: a row-basis of the rigidity
matrix at a sufficiently generic placement supports the desired Laman
subgraph.

We isolate the row-independence relation between an edge set and a
placement, then prove two facts about it: (a)~row-independence implies
the $(2, 3)$-sparsity required by the Laman definition (a classical
Maxwell-type counting argument; see Jord\'an~\cite{jordan2016} Lemma
1.3.1 for the modern presentation we follow); and (b)~a row-basis of
size $2|V| - 3$ exists whenever $G$ is generically rigid (rank lower
bound + linear-algebra extension). Combining (a) and (b) gives a
$(2, 3)$-tight spanning subgraph, i.e.\ a Laman subgraph.

The two facts together amount to the ``easy'' direction of an
identity due to Lov\'asz--Yemini~\cite{lovaszYemini1982}: in
dimension $2$, an edge set is row-independent at a generic placement
if and only if its underlying subgraph is $(2, 3)$-sparse. The harder
converse direction of that identity --- $(2, 3)$-sparse
$\Rightarrow$ row-independent --- is \emph{not} needed for the iff
and is not formalized.

\subsubsection{Row-independence in the rigidity matroid}

\begin{definition}[Row-independent edge set at a placement]
  \label{def:edgeSet-rowIndependent}
  \uses{def:rigidityMap}
  Let $G : \mathrm{SimpleGraph}\,V$, let $p : V \to \R^{d}$ be a
  framework, and let $I \subseteq E(G)$. We say that $I$ is
  \emph{row-independent at $p$} if the family of edge-rows
  $\bigl(\mathrm{RigidityMap}\,G\,p\,(\cdot)\,e\bigr)_{e \in I}$,
  viewed as linear functionals on $\mathrm{Framework}\,V\,d$, is
  linearly independent over $\R$.

  Equivalently, the composition of $\mathrm{RigidityMap}\,G\,p$ with
  the coordinate projection $E(G) \to \R \;\to\; I \to \R$ has full
  rank $|I|$.
\end{definition}

\begin{remark}
  The row-independent edge sets form the independent sets of a
  matroid on $E(G)$ --- the \emph{rigidity matroid} of $(G, p)$. We
  do not package it as a \texttt{Mathlib.Combinatorics.Matroid}
  object: the (\(\Rightarrow\)) direction of Laman's theorem
  needs only the independence relation and its two named facts
  below, and the $\mathrm{Matroid}$ packaging is reusable but
  off the critical path.
\end{remark}

\subsubsection{Rank of the rigidity map at a generically rigid placement}

\begin{lemma}[Rank lower bound at a generically rigid placement, dim $2$]
  \label{lem:rigidityMap-finrank-range-ge-of-isGenericallyRigid-two}
  \lean{SimpleGraph.rigidityMap_finrank_range_ge_of_isGenericallyRigid_two}
  \leanok
  \uses{def:isGenericallyRigid, lem:framework-finrank,
    def:isInfinitesimallyRigid}
  Let $V$ be finite and $G : \mathrm{SimpleGraph}\,V$. If $G$ is
  generically rigid in dimension $2$, there exists a framework
  $p : V \to \R^{2}$ with
  \[
    2 |V| \;\le\; \dim_{\R}\,\bigl(\mathrm{range}\,(\mathrm{RigidityMap}\,G\,p)\bigr) + 3.
  \]
\end{lemma}
\begin{proof}
  \leanok
  Pick an infinitesimally rigid framework $p$ witnessing generic
  rigidity. Rank--nullity for $\mathrm{RigidityMap}\,G\,p$ gives
  $\dim\,\mathrm{Framework}\,V\,2 = \dim\,\mathrm{range} + \dim\,\ker$;
  by \cref{lem:framework-finrank} the left-hand side equals $2 |V|$,
  and by the definition of infinitesimal rigidity in dim $2$ the
  nullity is at most $3$. (The lemma is the rank-half of the
  edge-count theorem \cref{thm:isGenericallyRigid-card-mul-le} with
  the edge-count bound \cref{lem:rigidityMap-finrank-range-le}
  replaced by the rank itself.)
\end{proof}

\begin{lemma}[Row-independent edge basis from rank lower bound]
  \label{lem:exists-rowIndependent-edge-basis}
  \uses{def:edgeSet-rowIndependent,
    lem:rigidityMap-finrank-range-ge-of-isGenericallyRigid-two,
    lem:rigidityMap-finrank-range-le}
  Let $V$ be finite, $G$ generically rigid in dimension $2$, and $p$
  a framework as in
  \cref{lem:rigidityMap-finrank-range-ge-of-isGenericallyRigid-two}.
  Then there exists $I \subseteq E(G)$ with $|I| = 2 |V| - 3$ which is
  row-independent at $p$.
\end{lemma}
\begin{proof}
  The edge-indexed rows of $\mathrm{RigidityMap}\,G\,p$ span its
  range, which has dimension at least $2 |V| - 3$
  (\cref{lem:rigidityMap-finrank-range-ge-of-isGenericallyRigid-two})
  and at most $|E(G)|$ (\cref{lem:rigidityMap-finrank-range-le}). Extract
  a basis from a spanning family of finite-dimensional rows --- the
  standard \texttt{LinearIndependent.extend} / \texttt{Basis.extend}
  construction --- and read $I$ as the edge labels indexing the chosen
  rows.
\end{proof}

\subsubsection{Trivial motions and the rank upper bound under affine spanning}

In dimension $2$, the kernel of the rigidity map always contains the
\emph{trivial Euclidean motions}: the $2$-dimensional space of
translations plus, at affinely-spanning placements, one additional
infinitesimal rotation, in the sense of Asimow--Roth's identification
of trivial infinitesimal motions with the kernel
of $R(K_{|V|}, p)$~\cite{asimowRoth1978}. The induced rank bound
$\rk \le 2|V| - 3$ is the converse of the rank lower bound and
underlies the $(2, 3)$-sparsity claim below.

\begin{lemma}[Three trivial motions kill the rigidity map at an affinely-spanning placement, dim $2$]
  \label{lem:trivialMotions-three-le-ker-of-affinelySpanning-two}
  \uses{def:rigidityMap}
  Let $V$ be finite with $|V| \ge 2$, let $G : \mathrm{SimpleGraph}\,V$,
  and let $p : V \to \R^{2}$ be a framework whose image affinely
  spans $\R^{2}$. Then
  $\dim_{\R}\,\ker\,(\mathrm{RigidityMap}\,G\,p) \ge 3$.
\end{lemma}
\begin{proof}
  Exhibit three linearly independent motions in the kernel: the two
  constant translations $p'_v = e_0$ and $p'_v = e_1$, and the
  infinitesimal rotation $p'_v = R(p_v - c)$ where $R$ is the
  $90^{\circ}$-rotation matrix and $c$ is any reference point.
  Each is killed by the rigidity map at \emph{every} edge by the
  entry formula $\langle p_u - p_v,\, p'_u - p'_v \rangle$ ---
  translations cancel in the difference $p'_u - p'_v$, and the
  rotation case reduces to $\langle p_u - p_v,\, R(p_u - p_v)
  \rangle = 0$. Linear independence of the three motions in
  $\mathrm{Framework}\,V\,2$ uses the affine-span hypothesis on $p$:
  if a linear combination $\alpha e_0 + \beta e_1 + \gamma R(p_v - c)$
  is zero at every $v$, then $\gamma = 0$ unless $p$ is contained in a
  line, and then $\alpha = \beta = 0$ from any single vertex.
\end{proof}

\begin{lemma}[Rank upper bound from affine spanning, dim $2$]
  \label{lem:rigidityMap-finrank-range-le-of-affinelySpanning-two}
  \uses{lem:trivialMotions-three-le-ker-of-affinelySpanning-two,
    lem:framework-finrank}
  Let $V$ be finite with $|V| \ge 2$, let $G$ be a graph on $V$, and
  let $p : V \to \R^{2}$ affinely span. Then
  \[
    \dim_{\R}\,\bigl(\mathrm{range}\,(\mathrm{RigidityMap}\,G\,p)\bigr) + 3
    \;\le\; 2 |V|.
  \]
\end{lemma}
\begin{proof}
  Combine
  \cref{lem:trivialMotions-three-le-ker-of-affinelySpanning-two} with
  rank--nullity and the framework-space dimension
  \cref{lem:framework-finrank}.
\end{proof}

\subsubsection{Sparsity from row-independence (Lov\'asz--Yemini, easy direction)}

The sparsity claim transports the rank upper bound from the induced
subframework $H[S]$ at $p|_{S}$ back to the row count of the
$H[S]$-rows in $\mathrm{RigidityMap}\,G\,p$. The two sides match
exactly because edges in $H[S]$ have both endpoints in $S$, so the
corresponding rows are supported on the columns indexed by $S$.
We need the witness placement to affinely span on every
size-$\ge 3$ subset; \cref{lem:exists-affinelySpanning-rigid-placement-two}
below provides one.

\begin{lemma}[Affinely-spanning rigid placement, dim $2$]
  \label{lem:exists-affinelySpanning-rigid-placement-two}
  \uses{def:isGenericallyRigid, lem:isInfinitesimallyRigid-eventually}
  Let $V$ be finite with $|V| \ge 2$ and $G$ generically rigid in
  dimension $2$. There exists a framework $p : V \to \R^{2}$ which is
  infinitesimally rigid for $G$ and whose restriction $p|_{S}$
  affinely spans $\R^{2}$ for every $S \subseteq V$ with $|S| \ge 3$.
\end{lemma}
\begin{proof}
  The set of infinitesimally rigid placements is open at the
  generic-rigidity witness by
  \cref{lem:isInfinitesimallyRigid-eventually}; the set of placements
  whose every size-$\ge 3$ restriction affinely spans is the
  complement of a finite union of hyperplanes (one ``three points
  collinear'' equation per triple in $V$), hence open and dense in
  $\mathrm{Framework}\,V\,2$. The two open sets meet on a
  neighborhood of the witness, giving a placement with both
  properties.
\end{proof}

\begin{lemma}[$(2, 3)$-sparsity from row-independence, dim $2$]
  \label{lem:isSparse-of-rowIndependent-two}
  \uses{def:isSparse, def:edgeSet-rowIndependent,
    lem:rigidityMap-finrank-range-le-of-affinelySpanning-two,
    lem:exists-affinelySpanning-rigid-placement-two}
  Let $V$ be finite, $G : \mathrm{SimpleGraph}\,V$, and $p$ a framework
  on $V$ in dimension $2$ such that $p|_{S}$ affinely spans for every
  $S \subseteq V$ with $|S| \ge 3$. If $I \subseteq E(G)$ is
  row-independent at $p$, then the spanning subgraph $H \le G$ with
  edge set $I$ is $(2, 3)$-sparse.
\end{lemma}
\begin{proof}
  Fix $S \subseteq V$. The edges of $H$ with both endpoints in $S$
  are exactly the edges of the induced subgraph $H[S]$; we show
  $|\edgesIn[H]{S}| + 3 \le 2 |S|$.

  For $|S| \le 1$ the proviso $3 \le 2|S|$ in the definition of
  $(2, 3)$-sparsity is false, so the bound holds vacuously. For
  $|S| = 2$ the induced subgraph has at most one edge, and
  $1 + 3 \le 4 = 2 |S|$.

  For $|S| \ge 3$: the rows of $\mathrm{RigidityMap}\,G\,p$ indexed
  by $\edgesIn[H]{S}$ are supported on the columns indexed by $S$
  (each row's nonzero entries are at the two endpoints of the
  corresponding edge, both of which lie in $S$). Hence row-independence
  of these rows in the full matrix is equivalent to row-independence
  in the column-restricted matrix, which is exactly the rigidity
  matrix $\mathrm{RigidityMap}\,(H[S])\,(p|_{S})$ (modulo identifying
  the $S$-indexed columns with $\mathrm{Framework}\,S\,2$). By
  \cref{lem:rigidityMap-finrank-range-le-of-affinelySpanning-two}
  applied to $H[S]$ at the affinely-spanning placement $p|_{S}$, the
  rank of that matrix is at most $2 |S| - 3$. Since row-independence
  forces the row count $|\edgesIn[H]{S}|$ to equal the rank,
  $|\edgesIn[H]{S}| \le 2 |S| - 3$.
\end{proof}

\subsubsection{Assembly}

\begin{theorem}[Every generically rigid graph contains a Laman subgraph]
  \label{thm:isGenericallyRigid-exists-isLaman-le}
  \lean{SimpleGraph.IsGenericallyRigid.exists_isLaman_le}
  \uses{def:isGenericallyRigid, def:isLaman, def:isTight,
    lem:rigidityMap-finrank-range-ge-of-isGenericallyRigid-two,
    lem:exists-rowIndependent-edge-basis,
    lem:isSparse-of-rowIndependent-two,
    lem:exists-affinelySpanning-rigid-placement-two}
  Let $V$ be finite with $|V| \ge 2$. If $G$ is generically rigid in
  dimension $2$, then there exists a Laman spanning subgraph
  $H \le G$.
\end{theorem}
\begin{proof}
  Pick a framework $p$ that is both infinitesimally rigid for $G$
  and affinely-spanning on every size-$\ge 3$ subset, via
  \cref{lem:exists-affinelySpanning-rigid-placement-two}. Apply
  \cref{lem:exists-rowIndependent-edge-basis} to extract
  $I \subseteq E(G)$ with $|I| = 2 |V| - 3$ row-independent at $p$.
  Let $H \le G$ be the spanning subgraph with edge set $I$. By
  \cref{lem:isSparse-of-rowIndependent-two}, $H$ is $(2, 3)$-sparse;
  together with $|E(H)| = 2 |V| - 3$ this makes $H$ $(2, 3)$-tight,
  i.e.\ Laman.
\end{proof}

\subsection{Laman's theorem}

\begin{theorem}[Laman's theorem]
  \label{thm:isGenericallyRigid-two-iff-exists-isLaman-le}
  \lean{SimpleGraph.isGenericallyRigid_two_iff_exists_isLaman_le}
  \uses{thm:isLaman-isGenericallyRigid-two,
    thm:isGenericallyRigid-exists-isLaman-le,
    lem:isGenericallyRigid-mono}
  Let $V$ be finite with $|V| \ge 2$ and $G : \mathrm{SimpleGraph}\,V$.
  Then $G$ is generically rigid in dimension $2$ iff $G$ contains a
  Laman spanning subgraph:
  \[
    G.\mathrm{IsGenericallyRigid}\,2
    \;\iff\;
    \exists H \le G,\; H.\mathrm{IsLaman}.
  \]
\end{theorem}
\begin{proof}
  The $\Rightarrow$ direction is
  \cref{thm:isGenericallyRigid-exists-isLaman-le} (Phase 6, not yet
  closed). For $\Leftarrow$: given $H \le G$ with $H$ Laman,
  \cref{thm:isLaman-isGenericallyRigid-two} gives
  $H.\mathrm{IsGenericallyRigid}\,2$, and monotonicity
  \cref{lem:isGenericallyRigid-mono} along $H \le G$ extends rigidity
  from $H$ to $G$.
\end{proof}
