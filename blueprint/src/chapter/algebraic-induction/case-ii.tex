\subsection{Case II: \texorpdfstring{$k > 0$}{k > 0} splitting (1-extension)}
\label{sec:molecular-algebraic-induction-caseII}

This subsection proves Katoh--Tanigawa's Case~II: a minimal $k$-dof-graph
with $k > 0$ and a reducible degree-$2$ vertex $v$ realizes its target
rank by re-inserting $v$, with its two hinges, onto a realization of the
smaller splitting-off graph $G_v^{ab}$ (KT Lemmas~6.7, 6.8) --- the
panel-hinge analogue of Whiteley's bar-joint $1$-extension. Two pieces of
shared infrastructure come first: the $+D$ rank-lift accounting for
pinning a single body (\cref{lem:case-II-rank-lift}) and the
edge-substitution relating the splitting-off graph to its parent
(\cref{lem:splitoff-edge-substitution}). The accounting itself
(\cref{lem:case-II}) is genericity-free; it assembles, together with the
inductive hypothesis, into the realization lemma
\cref{lem:case-II-realization}, which discharges KT Lemma~6.8 at every
deficiency $k > 0$. Genericity enters only through KT Claim~6.9, the
general-position choice of $v$'s two new hinges
(\cref{lem:exists-independent-panel-extensor}).

\begin{lemma}[Case II rank-lift accounting; the $+D$ core of KT Lemma 6.8]
  \label{lem:case-II-rank-lift}
  \lean{CombinatorialRigidity.Molecular.BodyHingeFramework.rankHypothesis_iff_finrank_pinnedMotions,
        CombinatorialRigidity.Molecular.PanelHingeFramework.rankHypothesis_withNormal_iff_finrank_pinnedMotions}
  \leanok
  \uses{def:rank-hypothesis, lem:rank-delete-vertex, def:framework-with-normal,
        lem:with-normal-preserves}
  In the codimension form of \cref{sec:molecular-rigidity-matrix}, pinning
  a body $v$ shifts the realization count by exactly $D$: a body-hinge
  framework $F$ on $G$ realizes the rank hypothesis at $k'$
  (\cref{def:rank-hypothesis}, $\dim Z(G,p) = D + k'$) \emph{iff} its
  body-$v$-pinned motion subspace has dimension $k'$. This is the
  algebraic $+D$ core of the panel-hinge $1$-extension:
  $\operatorname{pinned}(v)$ is the null space of the rigidity matrix with
  $v$'s $D$ columns deleted, and $\dim \operatorname{pinned}(v) + D = \dim
  Z(G,p)$ by the pin-a-body Lemma~5.1 (\cref{lem:rank-delete-vertex}). The
  same equivalence holds on the panel layer for a panel-hinge framework
  $P$ in which $v$ is unhinged, after choosing any panel normal for $v$
  (\cref{def:framework-with-normal}): replacing an unhinged body's normal
  changes no pinned-motion subspace (\cref{lem:with-normal-preserves}), so
  the extended framework realizes the rank hypothesis at $k'$ exactly when
  $P$ itself has body-$v$-pinned motion dimension $k'$.
\end{lemma}
\begin{proof}
  \leanok
  \uses{lem:rank-delete-vertex, lem:with-normal-preserves}
  Immediate from the pin-a-body identity $\dim \operatorname{pinned}(v) +
  D = \dim Z(G,p)$ (\cref{lem:rank-delete-vertex}): substitute it into the
  rank hypothesis $\dim Z(G,p) = D + k'$ and cancel the $D$ term. For the
  panel form, apply the body-level equivalence to the re-normaled
  framework, then identify its body-$v$-pinned motions with $P$'s own
  (\cref{lem:with-normal-preserves}, valid since $v$ is unhinged in $P$).
\end{proof}

\begin{lemma}[Edge-substitution bridge for splitting-off]
  \label{lem:splitoff-edge-substitution}
  \lean{Graph.removeVertex_le, Graph.removeVertex_le_splitOff,
        Graph.isLink_incident_of_not_removeVertex}
  \leanok
  The splitting-off $G_v^{ab}$ is \emph{not} a subgraph of $G$: it deletes
  the two $v$-incident edges $v a$, $v b$ but adds a fresh short-circuit
  edge $e_0 \notin E(G)$ joining $a$ and $b$. The two graphs are instead an
  \emph{edge substitution} of one another, sharing the common subgraph
  $G - v$ (all of $G$ away from $v$): $G - v \le G$ and
  $G - v \le G_v^{ab}$. The second inclusion holds because every edge of
  $G - v$ is an edge of $G$, hence $\neq e_0$, so it survives into
  $G_v^{ab}$. Conversely, every link of $G$ \emph{lost} by passing to
  $G - v$ is incident to $v$: the re-added edges over the common lower
  bound are exactly the $v$-incident ones, which is what lets the
  splitting-off accounting (\cref{lem:case-II}) collapse each re-added
  hinge constraint on a $v$-pinned motion to a single span membership at
  its non-$v$ endpoint.
\end{lemma}
\begin{proof}
  \leanok
  Both inclusions are the definition of $\le$ on multigraphs (vertex-set
  and link inclusion). For $G - v \le G_v^{ab}$, a link of $G - v$ is a
  link $e\,x\,y$ of $G$ with $x, y \neq v$; since $e \in E(G)$ and
  $e_0 \notin E(G)$ we have $e \neq e_0$, so $e$ lands in $G_v^{ab}$'s
  $v$-avoiding branch.
\end{proof}

\begin{lemma}[Case II: splitting off a reducible vertex; KT Lemmas 6.7, 6.8]
  \label{lem:case-II}
  \lean{CombinatorialRigidity.Molecular.BodyHingeFramework.isInfinitesimallyRigidOn_insert_iff,
        CombinatorialRigidity.Molecular.PanelHingeFramework.rankHypothesis_withNormal_withGraph_iff_finrank_pinnedMotions}
  \leanok
  \uses{def:rank-hypothesis, def:framework-with-normal, lem:reduction-step-pos,
        lem:case-II-rank-lift, lem:splitoff-edge-substitution,
        lem:exists-independent-panel-extensor}
  Let $G$ be a minimal $k$-dof-graph with $k > 0$ and a reducible degree-$2$
  vertex $v$ with neighbours $a$, $b$ (KT~4.6), and write
  $t = V(G) \setminus \{v\}$. For any body-hinge framework $F$ on $G$, $F$
  is rigid on $V(G)$ \emph{iff} $F$ is rigid on $t$ and every infinitesimal
  motion satisfies $Sv = Sw$ for each $w \in t$ --- the
  $1$-extension accounting stated purely in terms of $V(G)$.

  On the panel layer, restrict $F$ to the common subgraph $G - v$ of $G$
  and the splitting-off $G_v^{ab}$ (\cref{lem:splitoff-edge-substitution};
  $G_v^{ab}$ is a smaller minimal $(k-1)$-dof-graph,
  \cref{lem:reduction-step-pos}). Choosing any panel normal for the
  unhinged $v$ (\cref{def:framework-with-normal}) and passing back up to
  $G$ realizes the target rank at $k'$ (\cref{def:rank-hypothesis})
  \emph{iff} $F$ itself has body-$v$-pinned motion dimension $k'$
  (\cref{lem:case-II-rank-lift}), \emph{provided} every base-$v$-pinned
  motion of $F$ already clears the two new hinge constraints,
  $Sa \in \operatorname{span} C(e_a)$, $Sb \in \operatorname{span}
  C(e_b)$. That proviso is exactly KT Claim~6.9's general position,
  supplied by placing the two new supporting extensors in general
  position (\cref{lem:exists-independent-panel-extensor}).
\end{lemma}
\begin{proof}
  \leanok
  \uses{lem:case-II-rank-lift, lem:splitoff-edge-substitution,
        lem:exists-independent-panel-extensor}
  The $V(G)$-relative form is a direct case split on membership in
  $\operatorname{insert} v\ t$: both endpoints in $t$ reduce to rigidity
  of $F$ on $t$, and an endpoint equal to $v$ reduces to the hypothesis
  $Sv = Sw$. For the panel form, the $\le$ direction is unconditional ---
  re-adding $v$'s two edges over the common subgraph $G - v$ can only
  shrink the body-$v$-pinned motions (\cref{lem:splitoff-edge-substitution});
  the reverse $\ge$ is exactly the general-position hypothesis, since a
  base-$v$-pinned motion has $Sv = 0$, so each new edge's hinge constraint
  $Sv - Sw \in \operatorname{span} C(e)$ collapses to the span membership
  $Sw \in \operatorname{span} C(e)$ at its non-$v$ endpoint --- precisely
  what the two general-position conditions above supply. Composing with
  the rank-lift accounting (\cref{lem:case-II-rank-lift}) then reads the
  target rank off $F$'s own pinned dimension.
\end{proof}

\begin{lemma}[Case II realization: splitting off a degree-two vertex, $k > 0$; KT Lemma 6.8]
  \label{lem:case-II-realization}
  \lean{CombinatorialRigidity.Molecular.PanelHingeFramework.case_II_realization_all_k}
  \leanok
  \uses{def:rank-hypothesis, lem:reduction-step-pos,
        lem:rank-polynomial-of-le-finrank}
  Let $G$ be a $2$-edge-connected minimal $k$-dof-graph with $k > 0$ and no
  proper rigid subgraph, $|V(G)| \ge 3$. Given the induction hypothesis
  for every minimal $k'$-dof-graph with fewer vertices, $G$ carries a
  generic realization at rank $D(|V(G)|-1) - k$
  (\cref{def:rank-hypothesis}).
\end{lemma}
\begin{proof}
  \leanok
  \uses{lem:reduction-step-pos, lem:with-normal-preserves,
        lem:rank-polynomial-of-le-finrank, lem:genericity-device}
  By \cref{lem:reduction-step-pos}, $G$ has a reducible degree-$2$ vertex
  $v$ with distinct neighbours $a$, $b$ (the $2$-edge-connected,
  no-proper-rigid-subgraph hypotheses rule out the other two branches of
  the KT~4.6 trichotomy), and the splitting-off $G_v^{ab}$ is minimal
  $(k-1)$-dof with fewer vertices than $G$. The induction hypothesis
  supplies a generic realization $Q$ of $G_v^{ab}$ attaining rank
  $D(|V(G_v^{ab})|-1) - (k-1)$, with panel normals $n_a$, $n_b$ at $a$,
  $b$ linearly independent. Re-insert $v$ at the sheared normal
  $n_a + n_b$: since $v \notin V(G_v^{ab})$, this only assigns $v$'s
  (previously absent) normal and disturbs neither $Q$'s rigidity rows nor
  its motion space (\cref{lem:with-normal-preserves}). At this placement
  the new edge $vb$ reproduces the panel support extensor of $a, b$ up to
  sign and the new edge $va$ reproduces its negative, so a family of
  $D - 1$ rows independent through $v$'s screw column, together with the
  $D(|V(G_v^{ab})|-1) - (k-1)$ independent rows transported from $Q$
  through the common subgraph $G - v$, gives
  $D(|V(G)|-1) - k$ independent rigidity rows of $G$ at this placement ---
  the rank target exactly, with no residual gap to fill (unlike the
  deficiency-$0$ Case~III, where the analogous count is one row short).
  Unlike \cref{lem:case-II}'s bridge, this construction is direct and does
  not need the genericity-gated equality of the $1$-extension inclusion.
  The genericity device (\cref{lem:genericity-device}), via the
  rank-polynomial lift (\cref{lem:rank-polynomial-of-le-finrank}),
  re-realizes this rank at a general-position, algebraically-independent
  seed. One hypothesis rides explicitly: a fresh edge name for $G$ itself
  (an unused label outside $E(G)$, needed for the splitting-off's
  short-circuit edge $e_0$).
\end{proof}
