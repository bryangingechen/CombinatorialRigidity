\section{Henneberg moves}
\label{sec:henneberg}

Henneberg's theorem characterises Laman graphs as the graphs obtained
from $K_2$ by iterating two combinatorial moves --- the \emph{Type~I}
and \emph{Type~II} extensions described below. This section formalises
the two moves, proves each preserves the Laman property in both
directions, and develops the \emph{decomposition theorem}: every Laman
graph on $n \ge 3$ vertices is isomorphic to a Type~I or Type~II
extension of \emph{some Laman graph} $G'$ on $n - 1$ vertices
(\cref{thm:isLaman-exists-typeI-or-typeII-reverse}). This is the
inductive lever for the $\Leftarrow$ direction of Laman's theorem in
\cref{sec:laman-theorem}.

Throughout, the fresh vertex introduced by either move is encoded as
$\mathrm{none} : \mathrm{Option}\,V$, with the old vertices embedded
via $\mathrm{some}$. This is mathlib's canonical ``add one element''
idiom and avoids a freshness side-condition on every move.

\subsection{Definitions}

\begin{definition}[Type~I move]
  \label{def:typeI}
  \lean{SimpleGraph.Henneberg.typeI}
  \leanok
  Given $G : \mathrm{SimpleGraph}\,V$ and two vertices $a, b \in V$, the
  \emph{Type~I extension} $\mathrm{typeI}\,G\,a\,b$ is the graph on
  $\mathrm{Option}\,V$ obtained by adding a fresh vertex
  $w = \mathrm{none}$ adjacent to (the images under $\mathrm{some}$ of)
  exactly $a$ and $b$. The other adjacencies are those of $G$.
  Under $a \ne b$ the fresh vertex has degree~2; if $a = b$ the parallel
  edges collapse to a single one and the move preserves the Laman
  property only under $a \ne b$.
\end{definition}

\begin{definition}[Type~II move]
  \label{def:typeII}
  \lean{SimpleGraph.Henneberg.typeII}
  \leanok
  Given $G : \mathrm{SimpleGraph}\,V$ and three vertices $a, b, c \in V$,
  the \emph{Type~II extension} $\mathrm{typeII}\,G\,a\,b\,c$ is the
  graph on $\mathrm{Option}\,V$ obtained by deleting the edge
  $\{a, b\}$ from $G$ and adding a fresh vertex $w = \mathrm{none}$
  adjacent to $a$, $b$, and $c$.

  The move is well-defined for arbitrary $a, b, c$; it preserves the
  Laman property only under the four hypotheses
  $a \ne b$, $c \ne a$, $c \ne b$, and $\{a, b\} \in E(G)$.
\end{definition}

\subsection{Edge-set decomposition and counts}

The two moves admit a clean edge-set decomposition into ``old'' edges
(image of $E(G)$ under $\mathrm{some}$, with one edge removed in the
Type~II case) plus the two-or-three new fresh edges incident to
$\mathrm{none}$. The cardinality consequences feed the tightness
half of the preservation theorems below.

\begin{lemma}[Edge set of Type~I]
  \label{lem:typeI-edgeSet}
  \lean{SimpleGraph.Henneberg.typeI_edgeSet}
  \leanok
  \uses{def:typeI}
  \[
    E(\mathrm{typeI}\,G\,a\,b)
    \;=\;
    \mathrm{Sym2.map\;some}\,[E(G)]
    \;\cup\;
    \bigl\{\,
      \{\mathrm{none}, \mathrm{some}\,a\},
      \;
      \{\mathrm{none}, \mathrm{some}\,b\}
    \,\bigr\}.
  \]
\end{lemma}
\begin{proof}
  \leanok
  By case analysis on the two endpoints of an edge in
  $\mathrm{Option}\,V$, the only edges incident to $\mathrm{none}$
  are those joining $\mathrm{none}$ to $\mathrm{some}\,a$ or
  $\mathrm{some}\,b$ (by definition of $\mathrm{typeI}$); all other
  edges have both endpoints in $\mathrm{some}\,[V]$ and correspond
  bijectively to edges of $G$.
\end{proof}

\begin{lemma}[Edge count of Type~I]
  \label{lem:typeI-edgeSet-ncard}
  \lean{SimpleGraph.Henneberg.typeI_edgeSet_ncard}
  \leanok
  \uses{lem:typeI-edgeSet}
  If $V$ is finite and $a \ne b$, then
  \[
    |E(\mathrm{typeI}\,G\,a\,b)| \;=\; |E(G)| + 2.
  \]
\end{lemma}
\begin{proof}
  \leanok
  Use \cref{lem:typeI-edgeSet}: the two summands are disjoint (each new
  edge contains $\mathrm{none}$, which is not in the image of
  $\mathrm{some}$), $\mathrm{Sym2.map\;some}$ is injective, and the
  two new edges are distinct because $a \ne b$.
\end{proof}

\begin{lemma}[Edge set of Type~II]
  \label{lem:typeII-edgeSet}
  \lean{SimpleGraph.Henneberg.typeII_edgeSet}
  \leanok
  \uses{def:typeII}
  \[
    E(\mathrm{typeII}\,G\,a\,b\,c)
    \;=\;
    \mathrm{Sym2.map\;some}\bigl[E(G) \setminus \{\{a,b\}\}\bigr]
    \;\cup\;
    \bigl\{\,
      \{\mathrm{none}, \mathrm{some}\,a\}, \;
      \{\mathrm{none}, \mathrm{some}\,b\}, \;
      \{\mathrm{none}, \mathrm{some}\,c\}
    \,\bigr\}.
  \]
\end{lemma}
\begin{proof}
  \leanok
  Case analysis as in \cref{lem:typeI-edgeSet}: edges incident to
  $\mathrm{none}$ are exactly the three new ones, edges with both
  endpoints in $\mathrm{some}\,[V]$ correspond bijectively to edges of
  $G$, and the $\{a, b\}$ edge is explicitly removed by the
  $s(u, v) \ne s(a, b)$ clause in the definition.
\end{proof}

\begin{lemma}[Edge count of Type~II]
  \label{lem:typeII-edgeSet-ncard}
  \lean{SimpleGraph.Henneberg.typeII_edgeSet_ncard}
  \leanok
  \uses{lem:typeII-edgeSet}
  If $V$ is finite and $a \ne b$, $c \ne a$, $c \ne b$, then
  \[
    |E(\mathrm{typeII}\,G\,a\,b\,c)|
    \;=\;
    |E(G) \setminus \{\{a,b\}\}| + 3.
  \]
  We do not assume $\{a, b\} \in E(G)$: the deletion is recorded on
  the right-hand side rather than as a hypothesis.
\end{lemma}
\begin{proof}
  \leanok
  As in \cref{lem:typeI-edgeSet-ncard}, the two summands of
  \cref{lem:typeII-edgeSet} are disjoint and the
  $\mathrm{Sym2.map\;some}$ side is injective; the three new edges
  are pairwise distinct because $a, b, c$ are.
\end{proof}

\subsection{Preservation of the Laman property}

\begin{theorem}[Type~I preserves Laman]
  \label{thm:typeI-isLaman}
  \lean{SimpleGraph.Henneberg.typeI_isLaman}
  \leanok
  \uses{def:typeI, def:isLaman, lem:typeI-edgeSet-ncard}
  If $V$ is finite, $G$ is Laman, and $a \ne b$, then
  $\mathrm{typeI}\,G\,a\,b$ is Laman.
\end{theorem}
\begin{proof}
  \leanok
  Tightness follows from \cref{lem:typeI-edgeSet-ncard} and
  $|\mathrm{Option}\,V| = |V| + 1$: starting from
  $|E(G)| + 3 = 2|V|$, both sides gain $+2$.

  For sparsity, fix $s \subseteq \mathrm{Option}\,V$ with
  $3 \le 2|s|$ and write $s' := s \cap \mathrm{some}[V]$ for the
  $\mathrm{some}$-preimage. The edge set
  $\edgesIn[\mathrm{typeI}\,G\,a\,b]{s}$ decomposes as the $G$-edges
  on $s'$ plus the subset $T$ of the two new edges
  $\{\mathrm{none}, \mathrm{some}\,a\}$,
  $\{\mathrm{none}, \mathrm{some}\,b\}$ that lie in $s$. Case-split:
  \begin{itemize}
    \item If $\mathrm{none} \notin s$, then $T = \emptyset$ and
      $|s| = |s'|$, so the sparsity of $G$ at $s'$ transfers directly.
    \item If $\mathrm{none} \in s$, then $|s| = |s'| + 1$ and
      $|T| \le 2$. Sub-case on $|s'|$: if $|s'| = 0$ the precondition
      fails; if $|s'| = 1$, say $s' = \{w\}$, then no $G$-edge lies
      in $s'$ and $T \subseteq \{\{\mathrm{none}, \mathrm{some}\,w\}\}$
      so $|T| \le 1$, matching $2|s| - 3 = 0$; if $|s'| \ge 2$, apply
      sparsity of $G$ at $s'$ and use $|T| \le 2$.
  \end{itemize}
\end{proof}

\begin{theorem}[Type~II preserves Laman]
  \label{thm:typeII-isLaman}
  \lean{SimpleGraph.Henneberg.typeII_isLaman}
  \leanok
  \uses{def:typeII, def:isLaman, lem:typeII-edgeSet-ncard}
  If $V$ is finite, $G$ is Laman, the three vertices $a, b, c$ are
  pairwise distinct, and $\{a, b\} \in E(G)$, then
  $\mathrm{typeII}\,G\,a\,b\,c$ is Laman.
\end{theorem}
\begin{proof}
  \leanok
  Tightness: by \cref{lem:typeII-edgeSet-ncard} and
  $\{a, b\} \in E(G)$, the deletion drops the count by exactly $1$
  and the three new edges add $3$, so the move-graph has
  $|E(G)| + 2$ edges on $|V| + 1$ vertices. Then
  $\bigl(|E(G)| + 2\bigr) + 3 = 2(|V| + 1)$ follows from
  $|E(G)| + 3 = 2|V|$.

  Sparsity mirrors \cref{thm:typeI-isLaman}. The new wrinkle is that
  the deleted edge $\{a, b\}$ interacts with the (up to) three new
  fresh edges. Uniform $+2$ bound on the net gain: in the
  $\mathrm{none} \in s$, $|s'| \ge 2$ subcase, either
  $\{a, b\} \in \edgesIn{s'}$ (deletion loses $1$, the three new
  edges contribute $\le 3$, net $\le 2$) or $\{a, b\} \notin \edgesIn{s'}$
  (deletion loses $0$; since $\{a, b\} \in E(G)$ this forces
  $a \notin s'$ or $b \notin s'$, so the corresponding new edge
  $\{\mathrm{none}, \mathrm{some}\,a\}$ or
  $\{\mathrm{none}, \mathrm{some}\,b\}$ is also absent, bounding
  $|T'| \le 2$). The remaining sub-cases ($|s'| \in \{0, 1\}$ and
  $\mathrm{none} \notin s$) match \cref{thm:typeI-isLaman}.
\end{proof}

For the Type~I move, preservation is even tight: $\mathrm{typeI}\,G\,a\,b$
is Laman \emph{iff} $G$ is. The converse direction is used in
\cref{thm:isLaman-exists-typeI-or-typeII-reverse} below to peel a
Type~I extension back to its underlying graph.

\begin{theorem}[Type~I reflects Laman]
  \label{thm:typeI-reverse-isLaman}
  \lean{SimpleGraph.Henneberg.typeI_reverse_isLaman}
  \leanok
  \uses{def:typeI, def:isLaman, lem:typeI-edgeSet-ncard}
  If $V$ is finite, $a \ne b$, and $\mathrm{typeI}\,G\,a\,b$ is Laman,
  then $G$ is Laman.
\end{theorem}
\begin{proof}
  \leanok
  Dual to \cref{thm:typeI-isLaman}: lift an arbitrary
  $s' \subseteq V$ to $s := \mathrm{some}[s'] \subseteq \mathrm{Option}\,V$
  and apply sparsity of $\mathrm{typeI}\,G\,a\,b$ at $s$. With
  $\mathrm{none} \notin s$ the fresh-edge contribution collapses to
  zero, so the $\mathrm{typeI}$-sparsity bound at $s$ collapses to the
  $G$-sparsity bound at $s'$. Tightness transports via
  \cref{lem:typeI-edgeSet-ncard} run backwards.
\end{proof}

\begin{theorem}[Type~I is Laman iff its base is]
  \label{thm:typeI-isLaman-iff}
  \lean{SimpleGraph.Henneberg.typeI_isLaman_iff}
  \leanok
  \uses{thm:typeI-isLaman, thm:typeI-reverse-isLaman}
  If $V$ is finite and $a \ne b$, then
  $(\mathrm{typeI}\,G\,a\,b).\mathrm{IsLaman}
   \iff G.\mathrm{IsLaman}$.
\end{theorem}
\begin{proof}
  \leanok
  Combine \cref{thm:typeI-isLaman} and
  \cref{thm:typeI-reverse-isLaman}.
\end{proof}

\subsection{Decomposition: iso half}
\label{sec:henneberg-decomp}

A Laman graph on $n \ge 3$ vertices is, up to canonical relabelling,
the result of either a Type~I or a Type~II move applied to \emph{some}
graph $G'$ on $\{w \in V : w \ne v\}$, where $v$ is a chosen vertex of
degree $2$ or $3$. The statement below does \textbf{not} assert
$G'$ is itself Laman; the strengthened version
\cref{thm:isLaman-exists-typeI-or-typeII-reverse} does. We isolate the
iso-only version because the underlying construction is shared, and
the strengthened theorem reuses the iso witness it produces.

\begin{theorem}[Iso decomposition]
  \label{thm:isLaman-exists-typeI-or-typeII-iso}
  \lean{SimpleGraph.IsLaman.exists_typeI_or_typeII_iso}
  \leanok
  \uses{def:isLaman, def:typeI, def:typeII,
    lem:isLaman-exists-two-le-degree-le-three,
    lem:isLaman-exists-nonadj-among-three-neighbors}
  Let $G$ be Laman on a finite vertex type $V$ with $|V| \ge 3$. Then
  there exist $v \in V$ and a graph $G'$ on $\{w : V \mid w \ne v\}$
  such that either
  \begin{enumerate}
    \item there exist $a, b$ with $a \ne b$ and a graph isomorphism
      $G \simeq \mathrm{typeI}\,G'\,a\,b$; or
    \item there exist $a, b, c$ with $a \ne b$, $c \ne a$, $c \ne b$,
      $\{a, b\} \in E(G')$, and a graph isomorphism
      $G \simeq \mathrm{typeII}\,G'\,a\,b\,c$.
  \end{enumerate}
\end{theorem}
\begin{proof}
  \leanok
  By \cref{lem:isLaman-exists-two-le-degree-le-three} pick $v$ with
  $2 \le \deg_G(v) \le 3$, and let $G'$ be the induced subgraph
  $G[\{w \ne v\}]$ (augmented for the Type~II case below). The
  underlying type-level equivalence is
  $V \simeq \mathrm{Option}\,\{w : V \mid w \ne v\}$, mapping $v$ to
  $\mathrm{none}$. In the formalization the graph iso is built from
  this $\mathrm{Equiv}$ via a small custom constructor
  (\texttt{isoOfOptionSubtypeNe}), which case-splits on whether each
  endpoint equals $v$ and discharges the resulting four
  adjacency-rewrite obligations; the upgrade from a vertex
  $\mathrm{Equiv}$ to a graph iso is not automatic.

  \emph{Degree~$2$.} Let $\{a, b\} = N_G(v)$; then
  $N_G(v) = \{a, b\}$ and $a \ne b$, so the type-level equivalence
  upgrades to a graph iso $G \simeq \mathrm{typeI}\,G'\,a\,b$.

  \emph{Degree~$3$.} Let $\{a, b, c\} = N_G(v)$. By
  \cref{lem:isLaman-exists-nonadj-among-three-neighbors} at least one
  pair among $\{a, b\}, \{a, c\}, \{b, c\}$ is non-adjacent in $G$;
  relabel so this is $\{a, b\}$. Let $G'$ be the induced subgraph
  $G[\{w \ne v\}]$ \emph{augmented} with the bridging edge
  $\{a, b\}$. The type-level equivalence again gives a graph iso
  $G \simeq \mathrm{typeII}\,G'\,a\,b\,c$: the bridging edge in $G'$
  is exactly the one the Type~II definition removes.
\end{proof}

\subsection{Decomposition: Laman half}
\label{sec:henneberg-decomp-laman}

The Type~II branch of \cref{thm:isLaman-exists-typeI-or-typeII-iso}
makes a \emph{choice}: which non-adjacent neighbor pair of $v$ to
bridge with the new edge in $G'$. For arbitrary choices, the resulting
$G'$ need not be Laman; a 6-vertex counter-example forces a $K_4$
sub-structure on a bad bridge choice. The headline result of this
subsection (\cref{thm:isLaman-exists-typeI-or-typeII-reverse}) says
\emph{some} choice does produce a Laman $G'$. This is the
combinatorial heart of the $\Leftarrow$ direction of Laman's theorem
in \cref{sec:laman-theorem}.

The strategy is classical (Whiteley/Jord\'an): for each non-adjacent
pair $\{x, y\}$, either the corresponding $G'$ is Laman (return it),
or one obtains a \emph{tight blocker} --- a $(2, 3)$-tight subset of
$V$ disjoint from $v$ that contains both $x$ and $y$. Case analysis on
how many of the three pairs $\{a, b\}, \{a, c\}, \{b, c\}$ are
non-adjacent then combines the blockers via
\cref{thm:isTightOn-union-inter} into a tight subset of $V \setminus
\{v\}$ containing all three neighbors of $v$, which contradicts
sparsity of $G$ at $\{v\} \cup N_G(v)$.

\begin{theorem}[Per-pair tight blocker]
  \label{thm:isLaman-typeII-reverse-blocker}
  \lean{SimpleGraph.IsLaman.typeII_reverse_blocker}
  \leanok
  \uses{def:isLaman, def:typeII, def:isTightOn,
    lem:typeII-edgeSet-ncard, lem:isSparse-isTightOn-of-le,
    lem:isLaman-iso}
  Let $G$ be Laman with finite vertex type $V$. Suppose $v \in V$ has
  three pairwise distinct neighbors $\{x, y, c\}$, that $\{x, y\}$ is
  non-adjacent in $G$, and that the candidate $G'$ ---
  the induced subgraph on $\{w \in V : w \ne v\}$ augmented with the
  bridging edge $\{x, y\}$ --- is \textbf{not} Laman. Then there exists
  a finite $S \subseteq V$ with $v \notin S$, $x, y \in S$, and $S$
  $(2, 3)$-tight in $G$.
\end{theorem}
\begin{proof}
  \leanok
  By the iso half (the underlying construction of
  \cref{thm:isLaman-exists-typeI-or-typeII-iso}) we have an iso
  $G \simeq \mathrm{typeII}\,G'\,x\,y\,c$. Iso transport
  (\cref{lem:isLaman-iso}) carries $G$'s Laman property over to
  $\mathrm{typeII}\,G'\,x\,y\,c$; combining with
  \cref{lem:typeII-edgeSet-ncard} and
  $|\mathrm{Option}\,\{w \ne v\}| = |V|$ gives the global edge identity
  $|E(G')| + 3 = 2 |\{w \ne v\}|$. So the failure of $G'$ to be Laman
  is a failure of $(2, 3)$-sparsity: some finite
  $s' \subseteq \{w \ne v\}$ violates the bound. Lift to
  $S := \{w.\mathrm{val} : w \in s'\} \subseteq V \setminus \{v\}$. The
  edges of $G'$ on $s'$ correspond to edges of $G$ on $S$, plus
  possibly the bridge $\{x, y\}$ when $x, y \in s'$. (In the
  formalization, this "correspond to" is a named project-internal
  helper that gives the equality
  $\mathrm{Sym2.map}\;f \,[(G.\mathrm{comap}\;f).\edgesIn{s'}]
   = \edgesIn[G]{f[s']}$
  for $f = \mathrm{Subtype.val}$; the cardinality version it feeds
  needs injectivity of $\mathrm{Sym2.map}\;f$, which is itself a small
  lift of injectivity of $f$.) Case-split on $x, y \in s'$: in the
  bridge-included case, the $\le 1$ overcount yields a squeeze that
  forces $S$ tight in $G$ (\cref{lem:isSparse-isTightOn-of-le}); in
  the bridge-excluded case, the count violates $G$-sparsity directly,
  but the hypothesis $G.\mathrm{IsLaman}$ rules this out, so this
  branch is vacuous.
\end{proof}

\begin{theorem}[Strengthened decomposition: Laman half]
  \label{thm:isLaman-exists-typeI-or-typeII-reverse}
  \lean{SimpleGraph.IsLaman.exists_typeI_or_typeII_reverse}
  \leanok
  \uses{def:isLaman, def:typeI, def:typeII,
    thm:typeI-isLaman-iff,
    thm:isLaman-typeII-reverse-blocker,
    thm:isTightOn-union-inter,
    lem:isLaman-exists-two-le-degree-le-three,
    lem:isLaman-exists-nonadj-among-three-neighbors}
  Let $G$ be Laman on a finite vertex type $V$ with $|V| \ge 3$. Then
  there exist $v \in V$ and a \textbf{Laman} graph $G'$ on
  $\{w : V \mid w \ne v\}$ such that either
  \begin{enumerate}
    \item there exist $a, b$ with $a \ne b$ and a graph isomorphism
      $G \simeq \mathrm{typeI}\,G'\,a\,b$; or
    \item there exist $a, b, c$ with $a \ne b$, $c \ne a$, $c \ne b$,
      $\{a, b\} \in E(G')$, and a graph isomorphism
      $G \simeq \mathrm{typeII}\,G'\,a\,b\,c$.
  \end{enumerate}
\end{theorem}
\begin{proof}
  \leanok
  Pick $v$ with $2 \le \deg_G(v) \le 3$ as in
  \cref{thm:isLaman-exists-typeI-or-typeII-iso}.

  \emph{Degree~$2$.} The induced subgraph $G' = G[\{w \ne v\}]$ inherits
  Laman from $G$ by iso transport plus
  \cref{thm:typeI-isLaman-iff}: $\mathrm{typeI}\,G'\,a\,b$ is Laman
  (because $G$ is, by the underlying iso), so $G'$ itself is.

  \emph{Degree~$3$.} Let $N_G(v) = \{a, b, c\}$. For each pair among
  $\{a, b\}, \{a, c\}, \{b, c\}$ that is non-adjacent in $G$, the
  Type~II candidate built from that pair is either Laman (return it)
  or, by \cref{thm:isLaman-typeII-reverse-blocker}, yields a tight set
  $S \subseteq V \setminus \{v\}$ containing the pair. By
  \cref{lem:isLaman-exists-nonadj-among-three-neighbors} at least one
  pair is non-adjacent, so the all-adjacent leaf is vacuous; what
  remains is the contradiction step from the accumulated blockers,
  which splits on how many of the three pairs are non-adjacent.

  In each subcase the goal is the same: build a finite
  $T \subseteq V \setminus \{v\}$ containing $a, b, c$ with
  \[
    2|T| \;\le\; |\edgesIn{T}| + 3.
  \]
  Combined with $G$-sparsity at $T$ this forces $T$ to be
  $(2, 3)$-tight by \cref{lem:isSparse-isTightOn-of-le}. The set
  $\{v\} \cup T$ then carries the $|\edgesIn{T}|$ edges of $T$, plus
  the three edges $\{v, a\}, \{v, b\}, \{v, c\}$ from $v$ to its
  neighbors, and $G$-sparsity at $\{v\} \cup T$ is violated:
  $|\edgesIn{\{v\} \cup T}| \ge |\edgesIn{T}| + 3 = 2|T| = 2(|\{v\}
  \cup T| - 1) = 2|\{v\} \cup T| - 2$, exceeding the bound $2|\{v\}
  \cup T| - 3$.

  \begin{itemize}
    \item \emph{One non-adjacent pair.} Say $\{b, c\}$ is the only
      non-adjacent pair (so $\{a, b\}, \{a, c\} \in E(G)$). The
      blocker $S$ contains $b, c$ but possibly not $a$. If $a \in S$,
      take $T := S$ and the squeeze comes from tightness directly. If
      $a \notin S$, take $T := S \cup \{a\}$: the two edges
      $\{a, b\}, \{a, c\}$ lie in
      $\edgesIn{T} \setminus \edgesIn{S}$, so
      $|\edgesIn{T}| \ge |\edgesIn{S}| + 2$, and the squeeze at $T$
      follows from tightness of $S$ and $|T| = |S| + 1$.
    \item \emph{Two non-adjacent pairs.} Say $\{x, z\}$ and $\{y, z\}$
      are non-adjacent with shared neighbor $z$, so
      $\{x, y\} \in E(G)$. The blockers $S_1 \supseteq \{x, z\}$ and
      $S_2 \supseteq \{y, z\}$ both contain $z$. Sub-case on
      $|S_1 \cap S_2|$. If $\ge 2$, \cref{thm:isTightOn-union-inter}
      gives $T := S_1 \cup S_2$ tight, and $T$ contains $\{x, y, z\}$.
      If $= 1$, the intersection is exactly $\{z\}$, hence
      $x \notin S_2$ and $y \notin S_1$; the cross-edge $\{x, y\}$
      lies in $\edgesIn{T}$ but in neither $\edgesIn{S_1}$ nor
      $\edgesIn{S_2}$ for $T := S_1 \cup S_2$. Combining supermodularity
      (\cref{lem:edgesIn-ncard-add-le}) with $|S_1 \cap S_2| = 1$ (so
      $\edgesIn{S_1 \cap S_2} = \emptyset$) gives
      $|\edgesIn{S_1}| + |\edgesIn{S_2}| + 1 \le |\edgesIn{T}|$, and
      tightness of $S_1, S_2$ saturates the squeeze at $T$.
    \item \emph{Three non-adjacent pairs.} Three blockers
      $S_{ab}, S_{ac}, S_{bc}$. If any pairwise intersection has size
      $\ge 2$, \cref{thm:isTightOn-union-inter} applied to that pair
      gives a tight $T$ containing all three neighbors. Otherwise each
      pairwise intersection is the singleton of its shared neighbor
      ($S_{ab} \cap S_{ac} = \{a\}$, $S_{ab} \cap S_{bc} = \{b\}$,
      $S_{ac} \cap S_{bc} = \{c\}$). Two applications of supermodularity
      (\cref{lem:edgesIn-ncard-add-le}) bound $|\edgesIn{T}|$ from below
      by $|\edgesIn{S_{ab}}| + |\edgesIn{S_{ac}}| + |\edgesIn{S_{bc}}|$,
      using $\edgesIn{\{a\}} = \emptyset$ in the first step and
      $\edgesIn{\{b, c\}} = \emptyset$ (since $\{b, c\} \notin E(G)$)
      in the second; cardinality bookkeeping yields the squeeze at
      $T := S_{ab} \cup S_{ac} \cup S_{bc}$.
  \end{itemize}
\end{proof}

\subsection{Worked example: \texorpdfstring{$K_4 \setminus e$}{K4 minus e}}

\begin{theorem}[$K_4 \setminus e$ is Laman]
  \label{thm:top-fin-four-minus-edge-isLaman}
  \lean{SimpleGraph.top_fin_four_minus_edge_isLaman}
  \leanok
  \uses{thm:typeI-isLaman, thm:top-fin-two-isLaman, lem:isLaman-iso}
  The graph $K_4$ on the vertex set $\mathrm{Fin}\,4$, minus the single
  edge $\{2, 3\}$, is Laman.
\end{theorem}
\begin{proof}
  \leanok
  Apply \cref{thm:typeI-isLaman} twice from $K_2$
  (\cref{thm:top-fin-two-isLaman}): a single Type~I move yields a
  Laman graph on three vertices ($K_3$ in disguise), and a second
  Type~I move yields a Laman graph on four vertices. This Laman graph
  on $\mathrm{Option}\,(\mathrm{Option}\,(\mathrm{Fin}\,2))$ is iso to
  $K_4 \setminus \{2, 3\}$ on $\mathrm{Fin}\,4$ via an explicit
  bijection; transport along \cref{lem:isLaman-iso}.
\end{proof}
