\section{The combinatorial induction: graph operations and Theorem 4.9}
\label{sec:molecular-induction}

This chapter is the \textbf{Phase~20} dep-graph: the
\emph{combinatorial} half of Katoh--Tanigawa's proof of the molecular
conjecture --- \cite[\S3 (Lemmas 3.4, 3.5), \S4]{katohTanigawa2011},
the fourth stratum. Where \cref{sec:molecular-deficiency} built the
matroid $M(\tilde G)$, the $D$-deficiency, and the $k$-dof hierarchy,
this chapter develops the \emph{graph operations} that reduce a
minimal $k$-dof-graph to the two-vertex double edge, and assembles them
into Katoh--Tanigawa's Theorem~4.9. The algebraic induction of
Phases~21--23 then realizes each reduction step at the rigidity-matrix
rank. Throughout $D = \binom{d+1}{2} = \delta$ is the body-bar
dimension and $\tilde G = (D-1)\cdot G$ is the multiplied graph of
\cref{sec:body-hinge}.

\subsection{Inherited structural lemmas (KT 3.4 full form, KT 3.5)}
\label{sec:molecular-induction-inherited}

\begin{definition}[Vertex-induced subgraph from a fiber set]
  \label{def:induced-span}
  For a fiber set $X$ of the multiplied graph $\tilde G = (D-1)\cdot G$,
  the \emph{vertex span} $V(X)$ is the set of vertices of $G$ incident to
  some fiber of $X$ (\texttt{Graph.fiberSpan}), and the
  \emph{vertex-induced subgraph} $G[V(X)]$ is the induced subgraph of $G$
  on $V(X)$ (\texttt{Graph.inducedSpan}, mathlib's \texttt{Graph.induce}
  on $V(X)$). This is the induced-from-an-edge-set construction
  \cref{lem:circuit-induces-rigid} and the graph operations consume.
\end{definition}

\begin{lemma}[A circuit induces a rigid subgraph; KT Lemma 3.4, full form]
  \label{lem:circuit-induces-rigid}
  \lean{Graph.circuit_induces_isTight, Graph.circuit_induces_isRigidSubgraph}
  \leanok
  \uses{lem:circuit-rigid, def:induced-span, def:rigid-subgraph, thm:def-eq-corank}
  Let $X$ be a circuit of $M(\tilde G)$ and $e \in X$. Then $X - e$ is
  $(D,D)$-tight on its vertex span: $|X - e| = D(|V(X)| - 1)$, so $X - e$
  packs $D$ edge-disjoint spanning trees on $V(X)$ and the vertex-induced
  subgraph $G[V(X)]$ is a \emph{rigid subgraph} of $G$:
  $\operatorname{def}(\widetilde{G[V(X)]}) = 0$. This is the form
  Katoh--Tanigawa's Lemmas~4.5(i)/4.8 invoke (``Lemma~3.4 implies $G[V(X)]$
  is a (proper) rigid subgraph'').
\end{lemma}
\begin{proof}
  \leanok
  \uses{def:matroid-MG, lem:rank-matroidMG-le, lem:matroid-restrict-subgraph, thm:def-eq-corank}
  By \cref{lem:circuit-rigid}, $X - e$ is independent, hence $(D,D)$-sparse,
  giving the upper bound $|X - e| \le D(|V(X - e)| - 1) \le D(|V(X)| - 1)$.
  The matching lower bound $|X| > D(|V(X)| - 1)$ is forced by $X$ being a
  circuit: a circuit is a \emph{minimal} dependent set, so every proper
  subset of $X$ is independent (\,$(D,D)$-sparse\,), whence the sparsity
  failure of the dependent $X$ can only occur at $X$ itself
  (\cref{def:matroid-MG}, the boundary-regime cleanliness of $M(\tilde G)$).
  Together they give the tightness equality $|X - e| = D(|V(X)| - 1)$. To
  read off $\operatorname{def}(\widetilde{G[V(X)]}) = 0$: the rank of
  $M(\widetilde{G[V(X)]})$ is pinned to $D(|V(X)| - 1)$ from both sides ---
  the upper bound is the generic rank bound (\cref{lem:rank-matroidMG-le}),
  and the lower bound is the independent set $X - e$, which is independent in
  $M(\widetilde{G[V(X)]}) = M(\tilde G) \restriction E(\widetilde{G[V(X)]})$
  (\cref{lem:matroid-restrict-subgraph}, since every fiber of $X$ joins two
  vertices of $V(X)$, so $X \subseteq E(\widetilde{G[V(X)]})$) and has the
  tight size. The def\,$=$\,corank bridge (\cref{thm:def-eq-corank}) then
  forces the deficiency to $0$.
\end{proof}

\begin{lemma}[A rigid subgraph attains full rank; KT Lemma 3.5, rank core]
  \label{lem:rigid-full-rank}
  \lean{Graph.rank_matroidMG_of_isKDof_zero}
  \leanok
  \uses{def:rigid-subgraph, thm:def-eq-corank, def:matroid-MG}
  For a rigid subgraph $H$ (so $\operatorname{def}(\tilde H) = 0$) with
  $V(H) \ne \emptyset$ and $D \ge 1$, the matroid $M(\tilde H)$ has full
  rank $\operatorname{rank} M(\tilde H) = D(|V(H)| - 1)$. This is the rank
  quantity that contracting $H$ removes from both $\operatorname{rank}
  M(\tilde G)$ and the ambient $D(|V| - 1)$.
\end{lemma}
\begin{proof}
  \leanok
  Immediate from the def\,$=$\,corank bridge (\cref{thm:def-eq-corank},
  $\operatorname{rank} M(\tilde H) + \operatorname{def}(\tilde H) = D(|V(H)|
  - 1)$) with $\operatorname{def}(\tilde H) = 0$ from rigidity.
\end{proof}

\begin{lemma}[Contraction rank bridge; KT Lemma 3.5, contraction arithmetic]
  \label{lem:contract-rank-bridge}
  \lean{Graph.contract_matroidMG_rank}
  \leanok
  \uses{def:rigid-contraction, def:matroid-MG, lem:matroid-restrict-subgraph}
  For a subgraph $H \le G$, contracting the edge-fibers $E(\tilde H)$ in
  $M(\tilde G)$ removes exactly $\operatorname{rank} M(\tilde H)$:
  \[
    \operatorname{rank}\bigl(M(\tilde G) / E(\tilde H)\bigr)
      + \operatorname{rank} M(\tilde H)
      = \operatorname{rank} M(\tilde G).
  \]
\end{lemma}
\begin{proof}
  \leanok
  The standard matroid identity $\operatorname{rank}(M / C) =
  \operatorname{rank} M - \operatorname{rank}_M(C)$ in additive form, on
  $M = M(\tilde G)$ and $C = E(\tilde H)$. The restriction $M(\tilde G)
  \restriction E(\tilde H)$ is $M(\tilde H)$
  (\cref{lem:matroid-restrict-subgraph}), whose rank is
  $\operatorname{rank}_{M(\tilde G)}(E(\tilde H))$.
\end{proof}

\begin{lemma}[Contracting a rigid subgraph conserves the deficiency; KT Lemma 3.5]
  \label{lem:contract-deficiency-conservation}
  \lean{Graph.contract_matroidMG_deficiency_eq}
  \leanok
  \uses{lem:rigid-full-rank, lem:contract-rank-bridge, thm:def-eq-corank}
  For a rigid subgraph $H \le G$ (so $\operatorname{def}(\tilde H) = 0$) with $V(H) \ne
  \emptyset$ and $D \ge 1$, the corank of the matroid contraction $M(\tilde G) / E(\tilde
  H)$ at the reduced ambient $D(|V(G)| - |V(H)|)$ --- the trivial-motion count of the
  contracted graph, which has $|V(G)| - |V(H)| + 1$ vertices --- equals
  $\operatorname{def}(\tilde G)$:
  \[
    D\bigl(|V(G)| - |V(H)|\bigr) - \operatorname{rank}\bigl(M(\tilde G) / E(\tilde H)\bigr)
      = \operatorname{def}(\tilde G).
  \]
\end{lemma}
\begin{proof}
  \leanok
  Pure matroid bookkeeping. \Cref{lem:contract-rank-bridge} with the rank core
  \cref{lem:rigid-full-rank} ($\operatorname{rank} M(\tilde H) = D(|V(H)| - 1)$) gives
  $\operatorname{rank}(M(\tilde G) / E(\tilde H)) = \operatorname{rank} M(\tilde G) -
  D(|V(H)| - 1)$. Substituting into the def\,$=$\,corank bridge
  (\cref{thm:def-eq-corank}, $\operatorname{rank} M(\tilde G) + \operatorname{def}(\tilde
  G) = D(|V(G)| - 1)$) and cancelling the $D(|V(H)| - 1)$ between the rank drop and the
  ambient drop leaves $\operatorname{def}(\tilde G)$. No graph$\leftrightarrow$matroid
  correspondence is needed --- the statement is against the matroid contraction directly.
\end{proof}

\begin{lemma}[Minimality transports along a contraction; KT Lemma 3.5, second half]
  \label{lem:contract-minimality-transport}
  \lean{Graph.contract_minimality_transport}
  \leanok
  \uses{lem:subgraph-minimality, def:k-dof, def:rigid-contraction}
  Let $G$ be a minimal $k$-dof-graph and $H$ a subgraph. Then every base
  $B'$ of the matroid contraction $M(\tilde G) / E(\tilde H)$ meets every
  \emph{surviving} edge-fiber $\tilde e$ of an edge $e \in E(G) \setminus
  E(H)$: $B' \cap \tilde e \ne \emptyset$. This is the contraction analogue
  of the restriction transport \cref{lem:subgraph-minimality}.
\end{lemma}
\begin{proof}
  \leanok
  \uses{def:matroid-MG}
  A base $B'$ of $M(\tilde G) / E(\tilde H)$ lies in the contraction's
  ground $E(\tilde G) \setminus E(\tilde H)$, so it is disjoint from
  $E(\tilde H)$. Pick an $M(\tilde G)$-basis $J$ of $E(\tilde H)$; then
  $B' \cup J$ is a base of $M(\tilde G)$ (the deleted part $E(\tilde H)
  \setminus J$ consists of loops of the contraction $M(\tilde G) / J$, so
  $B'$ is a base there and \cref{def:matroid-MG}'s independence lifts). By
  $G$'s minimality, $(B' \cup J) \cap \tilde e \ne \emptyset$; the surviving
  fiber $\tilde e$ ($e \notin E(H)$) is disjoint from $E(\tilde H) \supseteq
  J$, so the witness lands in $B'$. No graph$\leftrightarrow$matroid
  correspondence is needed.
\end{proof}

\begin{lemma}[Rigid-subgraph contraction preserves minimality; KT Lemma 3.5]
  \label{lem:contraction-minimality}
  \lean{Graph.contraction_isMinimalKDof}
  \leanok
  \uses{def:rigid-subgraph, lem:subgraph-minimality, def:k-dof, lem:rigid-full-rank, lem:contract-rank-bridge, lem:contract-deficiency-conservation, lem:contract-minimality-transport}
  Let $G$ be a minimal $k$-dof-graph and let $H$ be a proper rigid
  subgraph of $G$. Then the contraction $G / E(H)$ (collapse $V(H)$ to
  a single vertex, keeping the edges of $G$ incident to $V(H)$ from
  outside) is again a minimal $k$-dof-graph, with the same deficiency.
  Stated on the matroid side --- the matroid contraction $M(\tilde G) /
  E(\tilde H)$ is a minimal $k$-dof matroid at the reduced ambient
  $D(|V(G)| - |V(H)|)$ --- so no graph$\leftrightarrow$matroid
  correspondence is needed. This is the engine of Case~I of the
  algebraic induction.
\end{lemma}
\begin{proof}
  \leanok
  \uses{lem:contract-deficiency-conservation, lem:contract-minimality-transport}
  The deficiency is unchanged by
  \cref{lem:contract-deficiency-conservation}: contracting a rigid
  subgraph removes $D(|V(H)| - 1)$ from both the rank of $M(\tilde G)$
  and the ambient $D(|V| - 1)$, so the corank --- hence the deficiency
  --- is conserved. Minimality (every base of the contracted matroid
  meets every edge-fiber) transports from $G$ along the contraction,
  using the subgraph-minimality machinery of
  \cref{lem:subgraph-minimality}.
\end{proof}

\subsection{Graph operations}
\label{sec:molecular-induction-operations}

\begin{definition}[Vertex removal, splitting-off, and edge-splitting]
  \label{def:graph-operations}
  \lean{Graph.removeVertex, Graph.splitOff, Graph.edgeSplit}
  \leanok
  \uses{def:k-dof}
  For a multigraph $G$ and a vertex $v$ of degree~2 with the two
  incident edges joining $v$ to $a$ and $b$:
  \begin{itemize}
    \item \emph{Removal} $G_v := G - v$ deletes $v$ and its incident
      edges.
    \item \emph{Splitting-off} $G_v^{ab}$ deletes $v$ and adds the new
      edge $ab$ (the two edges through $v$ are short-circuited into a
      single edge between its neighbours).
    \item \emph{Edge-splitting} $H_{ab}^v$ is the inverse: subdivide an
      edge $ab$ of $H$ by a fresh degree-2 vertex $v$, replacing $ab$
      with the path $a$--$v$--$b$. It satisfies $(H_{ab}^v)_v^{ab} = H$.
  \end{itemize}
\end{definition}

\begin{definition}[Rigid-subgraph contraction]
  \label{def:rigid-contraction}
  \lean{Graph.rigidContract}
  \leanok
  \uses{def:rigid-subgraph}
  For a rigid subgraph $H \le G$, the \emph{contraction} $G / E(H)$
  collapses the vertex set $V(H)$ to a single vertex and discards the
  edges of $H$, retaining every other edge of $G$ (with endpoints in
  $V(H)$ redirected to the collapsed vertex). On the matroid side this
  is the matroid contraction $M(\tilde G) / E(\tilde H)$ restricted to
  the surviving fibers.
\end{definition}

\subsection{Forest surgery (deferred --- see the remark on KT Lemma 4.1)}
\label{sec:molecular-induction-forest}

\begin{remark}[On Katoh--Tanigawa Lemma 4.1]
  \label{rem:kt-lemma-41}
  Formalizing the forest surgery of \cite[Lemma 4.1]{katohTanigawa2011}
  (p.~660; the 2009 arXiv predecessor, Lemma 5.1, p.~11, is essentially
  verbatim) surfaced a three-layer subtlety. \emph{The molecular
  conjecture and Katoh--Tanigawa's proof are not in question}; the value
  is in stating exactly which layer is which.
  \begin{enumerate}
    \item \emph{Statement-as-quantified --- false.} Both versions read
      ``for \emph{any} independent set $I$ of $M(\tilde G)$ \dots there
      exists $I'$ \dots with $|I'| = |I| - D$''. For any independent $I$
      with $|I| < D$ --- e.g.\ $I = \emptyset$ --- this demands $|I'| =
      |I| - D < 0$, which is impossible. The intended quantifier is
      clearly over \emph{bases} of $M(\tilde G)$; the universal-over-$I$
      reading must be restricted. We record the literal disproof as
      \cref{ex:kt-41-overquantified}.
    \item \emph{Proof of the intended (base) case --- an unjustified
      step.} ``Convert each $F_i$ to $F'_i$ with $|F'_i| = |F_i| - 1$
      for each $i$'' (2011), and the equation $2h' + (D - h') = h$
      (2009), silently assume the chosen $D$-forest packing is
      \emph{balanced at $v$}: every forest uses at least one edge at the
      degree-2 vertex $v$. Neither version justifies this, and for a
      general packing it can fail (pigeonhole when $h < D$); when it
      fails, both the cardinality count and the slack clause break.
      Whether the balanced packing is achievable for a base we leave
      \emph{open} --- Katoh--Tanigawa omit a justification and we did not
      recover one.
    \item \emph{Intended content --- true.} What the induction
      (\cref{lem:dof-tracking}) actually consumes is the deficiency
      inequality $\operatorname{def}(\widetilde{G_v^{ab}}) \le
      \operatorname{def}(\tilde G)$. This is correct, and we establish it
      directly (\cref{lem:splitoff-deficiency}) by a partition-count
      comparison through the def\,$=$\,corank bridge
      (\cref{thm:def-eq-corank}), bypassing the forest surgery entirely.
  \end{enumerate}
  Accordingly the forest surgery (both the splitting-off
  \cref{lem:forest-surgery-split} and the edge-splitting inverse
  \cref{lem:forest-surgery-unsplit}) is \textbf{deferred / off the
  Theorem-4.9 critical path}: the substrate of the splitting-off direction
  is formalized but completes only if the balanced-packing assumption of
  layer~(2) is settled, and the edge-splitting inverse, while sound, is not
  needed by Theorem~4.9 either. Both the splitting-off deficiency bound
  (\cref{lem:splitoff-deficiency}, KT~4.3) \emph{and} the removal bound
  (\cref{lem:removal-deficiency}, KT~4.4) are reached instead through the
  deficiency-counting route of \cref{sec:molecular-induction-dof}, on the
  green \cref{thm:def-eq-corank} infrastructure --- the removal bound by
  exactly the same partition-count comparison as the splitting-off bound,
  needing only $D \ge 2$ (\cref{rem:kt-lemma-44}).
\end{remark}

\begin{example}[KT Lemma 4.1's universal quantification is unsatisfiable]
  \label{ex:kt-41-overquantified}
  % Formalized as an unnamed Lean `example`; no `\lean{}` pointer to pin.
  \leanok
  \uses{def:matroid-MG, rem:kt-lemma-41}
  Taking $I = \emptyset$ (independent in any matroid), the promised $I'$
  with $|I'| = |I| - D = -D$ would need $|I'| + D = 0$ in $\mathbb{N}$,
  impossible for $D = \binom{d+1}{2} \ge 1$. This is layer~(1) of
  \cref{rem:kt-lemma-41}: the universal-over-$I$ quantification is
  formally false, witnessing that the intended quantifier ranges over
  bases. Formalized as a one-line \texttt{example} in
  \texttt{Molecular/Induction.lean}.
\end{example}

\begin{lemma}[Forest-packing decomposition of an $M(\tilde G)$-independent set]
  \label{lem:forest-packing-decomp}
  \lean{Graph.matroidMG_indep_iff_exists_forest_packing}
  \leanok
  \uses{def:matroid-MG}
  A fiber set $I \subseteq E(\tilde G)$ is independent in $M(\tilde G)$ if and only
  if it is covered by $D = \binom{d+1}{2}$ cycle-matroid-independent fiber sets
  $F_0, \dots, F_{D-1}$ --- the \emph{$D$ edge-disjoint forests on $V(\tilde G)$} of
  Katoh--Tanigawa's surgery: $\bigcup_{i} F_i = I$ with each $F_i$ a forest of
  $\tilde G$. This is the matroidal reading of ``$I$ partitions into $D$ forests'',
  fixing the framing of the forest surgery below: a \emph{forest} is a
  cycle-matroid-independent fiber set, and the $D$-forest partition is the union
  decomposition of $M(\tilde G)$ --- no hand-rolled acyclicity predicate.
\end{lemma}
\begin{proof}
  \leanok
  $M(\tilde G)$ is the $D$-fold cycle-matroid union restricted to $E(\tilde G)$
  (\cref{def:matroid-MG}), so independence unfolds through matroid restriction and
  the matroid-union characterization (Nash--Williams~\cite{nashWilliams1966},
  Edmonds~\cite{edmonds1965}): independence in $\bigcup_{i < D}
  \operatorname{cycleMatroid}(\tilde G)$ is exactly a cover by $D$
  cycle-matroid-independent sets.
\end{proof}

\begin{lemma}[Forest surgery at a degree-2 vertex, splitting-off direction; KT Lemma 4.1]
  \label{lem:forest-surgery-split}
  \uses{def:graph-operations, def:matroid-MG, thm:def-eq-corank, lem:forest-packing-decomp, rem:kt-lemma-41}
  \emph{Deferred / off the Theorem-4.9 critical path}
  (\cref{rem:kt-lemma-41}): its substrate is formalized in
  \texttt{Molecular/Induction.lean}, but the lemma completes only if the
  balanced-packing assumption of \cref{rem:kt-lemma-41}~(2) is settled;
  the induction instead uses \cref{lem:splitoff-deficiency}.
  Let $v$ be a degree-2 vertex of $G$ with neighbours $a, b$. A
  packing of $\tilde G$ into $D$ edge-disjoint forests, restricted to
  the fibers not incident to $v$ and rerouted across $v$, induces a
  packing of $\widetilde{G_v^{ab}}$, with the deficiency changing in a
  controlled way. Concretely the surgery rewires the $D-1$ fibers of
  one of $v$'s edges through $v$ onto the new edge $ab$.
\end{lemma}
\begin{proof}
  Explicit forest surgery: each of the $D$ spanning forests of $\tilde
  G$ uses some subset of the $2(D-1)$ fibers at $v$; rerouting a
  $v$-traversing tree-path across the short-circuit $ab$ keeps each
  forest acyclic and edge-disjoint. The bookkeeping is the hard new
  combinatorial core of Phase~20; no existing analogue in the project
  or mathlib.
\end{proof}

\begin{lemma}[Forest surgery at a degree-2 vertex, edge-splitting direction; KT Lemma 4.2]
  \label{lem:forest-surgery-unsplit}
  \uses{def:graph-operations, lem:forest-packing-decomp}
  \emph{Deferred / off the Theorem-4.9 critical path.} Unlike
  \cref{lem:forest-surgery-split}, this direction is \textbf{sound} --- it
  carries no balanced-packing gloss --- but it is not needed by
  Theorem~4.9: the removal bound it was thought to gate
  (\cref{lem:removal-deficiency}, KT~4.4) is in fact reached by the same
  deficiency-count route as \cref{lem:splitoff-deficiency}
  (\cref{rem:kt-lemma-44}). The inverse direction: a packing of
  $\widetilde{H}$ into $D$ forests on $V$ extends across an edge-split
  $H_{ab}^v$ to a packing of $\widetilde{H_{ab}^v}$ into $D$ forests on $V
  \cup \{v\}$, matching deficiencies under $(H_{ab}^v)_v^{ab} = H$.
\end{lemma}
\begin{proof}
  The fresh vertex $v$ starts isolated, so adding a single $v$-incident
  fiber to a forest on $V$ can never close a cycle. KT~4.2(i) at $h' = 0$
  thus adds one $va$-fiber to each of $D-1$ forests and one $vb$-fiber to
  the last, lifting the packing of $\widetilde{H}$ to one of
  $\widetilde{H_{ab}^v}$ with $|I| = |I'| + D$ --- with \emph{no}
  rerouting across $v$ and \emph{no} appeal to the balanced-packing
  assumption that obstructs \cref{lem:forest-surgery-split}. (The general
  $h' > 0$ case does reroute and shares the splitting-off surgery's
  subtleties.) This is the lift Katoh--Tanigawa use to derive KT~4.4; we
  reach KT~4.4 by deficiency counting instead, leaving this lemma
  unformalized.
\end{proof}

\subsection{Degree-of-freedom tracking (via deficiency counting)}
\label{sec:molecular-induction-dof}

The dof-conservation laws Katoh--Tanigawa derive from the forest surgery
(KT~4.3--4.5) are obtained here directly from the deficiency
max-formula, on the green infrastructure of
\cref{sec:molecular-deficiency} --- the partition deficiency
$\operatorname{def}(\tilde G) = \max_P [\, D(|P| - 1) - (D - 1)\,d_G(P)\,]$
and the def\,$=$\,corank bridge (\cref{thm:def-eq-corank}). This is the
route that bypasses the deferred forest surgery
(\cref{rem:kt-lemma-41}).

\begin{lemma}[Splitting-off changes the deficiency by at most one; KT Lemma 4.3]
  \label{lem:splitoff-deficiency}
  \lean{Graph.splitOff_deficiency_le, Graph.splitOff_deficiency_ge}
  \leanok
  \uses{def:graph-operations, def:matroid-MG, thm:def-eq-corank}
  Let $v$ be a degree-2 vertex of $G$ with neighbours $a, b$, carried by
  the two edges $eₐ$ (joining $v, a$) and $e_b$ (joining $v, b$). Then
  \[
    \operatorname{def}(\tilde G) - 1
      \;\le\; \operatorname{def}(\widetilde{G_v^{ab}})
      \;\le\; \operatorname{def}(\tilde G),
  \]
  so $G_v^{ab}$ is either a $k$-dof-graph or a $(k-1)$-dof-graph
  (KT~4.3(i)). This two-sided bound is the substantive splitting-off fact
  the induction consumes, replacing the forest surgery of
  \cref{lem:forest-surgery-split}. The matroid-base characterization of
  \emph{which} alternative holds (KT~4.3(ii): $G_v^{ab}$ is $k$-dof iff
  some base $B'$ of $M(\widetilde{G_v^{ab}})$ has $|\tilde{ab} \cap B'| <
  D - 1$) is Katoh--Tanigawa's reading through the deferred forest surgery
  (\cref{rem:kt-lemma-41}) and is not needed for Theorem~4.9.
\end{lemma}
\begin{proof}
  \leanok
  Both bounds are per-partition deficiency-count comparisons, on the green
  \texttt{deficiency} infrastructure of \cref{sec:molecular-deficiency} and
  \emph{no forests}. For the upper bound, each partition $P'$ of $V - v$
  extends to a partition $P$ of $V$ by dropping $v$ into $a$'s block ($f =
  \mathrm{update}\,f'\,v\,(f'\,a)$); then $|P| = |P'|$, since $v$'s new
  label $f'(a)$ is already carried by $a \in V - v$. The edge $va$ no longer
  crosses $P$ (both endpoints carry $f'(a)$), $vb$ crosses $P$ exactly when
  the short-circuit $ab$ crosses $P'$, and every other edge survives with
  the same crossing status; so $d_G(P) \le d_{G_v^{ab}}(P')$ via the
  injection $e_b \mapsto ab$, and the per-partition deficiency does not
  decrease.

  For the lower bound, take a partition $P$ of $V$ attaining
  $\operatorname{def}(\tilde G)$ (a finite supremum) and reuse the same
  labeling on $V - v$. If $v$'s label is shared by another vertex, $|P|$ is
  unchanged and the crossing count does not increase ($va, vb$ leave, $ab$
  enters), so $\operatorname{def}_{G_v^{ab}}(P) \ge \operatorname{def}(\tilde
  G)$. If $v$ is isolated in its block, $|P|$ drops by exactly one but both
  $va$ and $vb$ crossed and $ab$ enters, so the crossing count drops by at
  least one and $\operatorname{def}_{G_v^{ab}}(P) \ge D(|P|-2) -
  (D-1)(d_G(P)-1) = \operatorname{def}(\tilde G) - 1$. Either way
  $\operatorname{def}(\widetilde{G_v^{ab}}) \ge \operatorname{def}(\tilde
  G) - 1$.
\end{proof}

\begin{remark}[KT Lemma 4.4's lower bound \emph{is} a deficiency-counting fact, via the isolated-case crossing-drop]
  \label{rem:kt-lemma-44}
  \uses{lem:splitoff-deficiency, lem:removal-deficiency}
  Katoh--Tanigawa derive KT~4.4 ($\operatorname{def}(\widetilde{G_v}) \ge
  k$, p.~662) from the $h' = 0$ case of the unsplit forest surgery
  (Lemma~4.2(i)): a base $B'$ of $M(\widetilde{G_v})$, read as $D$ forests
  on $V - v$, lifts to $D$ forests on $V$ by adding one fresh $v$-edge per
  forest ($v$ is isolated, so no added edge closes a cycle), giving an
  independent set of $M(\tilde G)$ of size $|B'| + D$. That derivation is
  sound. \emph{But the forest surgery is not necessary}: KT~4.4 falls to
  the same partition-count comparison that carries
  \cref{lem:splitoff-deficiency}, with no forests.

  Take a partition $P$ of $V$ attaining $\operatorname{def}(\tilde G) = k$
  and reuse the labeling on $V - v$. The crossing edges of $G_v$ are
  exactly those of $G$ minus the $v$-incident ones; since $v$ has degree
  two ($e_a, e_b$ the only $v$-edges), $d_{G_v}(P) = d_G(P) - c$ with $c
  \in \{0, 1, 2\}$. If $v$'s block is shared, $|P|$ is unchanged and
  $\operatorname{def}_{G_v}(P) = k + (D-1)c \ge k$ --- the dropped
  $v$-edges \emph{help}, because $\operatorname{partitionDef}$ carries
  $-(D-1)\,d$. If $v$ is isolated in its block, $|P|$ drops by one, but
  then $a, b$ lie in blocks distinct from $v$'s, so \emph{both} $e_a$ and
  $e_b$ cross ($c = 2$), giving $\operatorname{def}_{G_v}(P) = k - D +
  2(D-1) = k + (D-2) \ge k$. The crossing-drop $2(D-1)$ pays for the
  part-loss $D$ exactly because $D \ge 2$.

  This corrects an earlier analysis (\texttt{notes/Phase20.md}
  ``Finding~2'') that had concluded KT~4.4 was \emph{not} a
  deficiency-counting fact, on the grounds that restriction
  rank-monotonicity gives only $\operatorname{rank} M(\tilde G) -
  \operatorname{rank} M(\widetilde{G_v}) \le 2(D-1)$ and a single-partition
  comparison only $\ge k - D$. The flaw was overlooking that in the
  \emph{isolated} case both $v$-edges are \emph{forced} to cross, so the
  crossing-drop is exactly $2(D-1)$, not merely ``$\le 2$ in the wrong
  direction'': with the sign $-(D-1)\,d$ in $\operatorname{partitionDef}$,
  a larger crossing-drop \emph{raises} the deficiency, and $D \ge 2$ makes
  it outweigh the lost part. KT~4.4 thus needs neither the forest surgery
  nor a rank-increment existence statement --- only $D \ge 2$, trivially
  true in the molecular regime ($n \ge 2 \Rightarrow D = \binom{n+1}{2}
  \ge 3$). Katoh--Tanigawa's forest-surgery route remains a valid
  alternative; the deficiency count is simply shorter.
\end{remark}

\begin{lemma}[Vertex removal raises the deficiency; KT Lemma 4.4]
  \label{lem:removal-deficiency}
  \lean{Graph.removeVertex_deficiency_ge}
  \leanok
  \uses{def:graph-operations, lem:splitoff-deficiency}
  Let $v$ be a degree-2 vertex of $G$ with neighbours $a, b$, carried by
  the two edges $e_a$ (joining $v, a$) and $e_b$ (joining $v, b$), and let
  $D = \binom{d+1}{2} \ge 2$. Then vertex removal does not decrease the
  deficiency, $\operatorname{def}(\tilde G) \le
  \operatorname{def}(\widetilde{G_v})$; in particular if $G$ is a
  $k$-dof-graph then $\operatorname{def}(\widetilde{G_v}) \ge k$. This is
  the other half of the local dof bookkeeping at $v$. (Katoh--Tanigawa's
  additional clause --- if equality holds there is a base $B$ of $M(\tilde
  G)$ with $|\tilde{vb} \cap B| = 1$ --- is read through the forest surgery
  and is not needed for Theorem~4.9.)
\end{lemma}
\begin{proof}
  \leanok
  By the same per-partition deficiency-count comparison as
  \cref{lem:splitoff-deficiency}, \emph{no forests}
  (\cref{rem:kt-lemma-44}). Removal is simpler than splitting-off: $G_v =
  G - v$ adds no fresh short-circuit edge, so the crossing count strictly
  drops. Take a partition $P$ of $V$ attaining $\operatorname{def}(\tilde
  G)$ and reuse the labeling on $V - v$. If $v$'s block is shared, $|P|$ is
  unchanged and the crossing count does not increase, so
  $\operatorname{def}_{G_v}(P) \ge \operatorname{def}(\tilde G)$. If $v$ is
  isolated, $|P|$ drops by one but both $e_a, e_b$ cross (their other
  endpoints $a, b$ lie in blocks distinct from $v$'s), so the crossing
  count drops by exactly $2$ and $\operatorname{def}_{G_v}(P) = D(|P| - 2)
  - (D-1)(d_G(P) - 2) = \operatorname{def}(\tilde G) + (D - 2) \ge
  \operatorname{def}(\tilde G)$ using $D \ge 2$. Either way
  $\operatorname{def}(\widetilde{G_v}) \ge \operatorname{def}(\tilde G)$.
\end{proof}

\begin{lemma}[Deficiency under vertex removal and splitting-off; KT Lemmas 4.3--4.5]
  \label{lem:dof-tracking}
  \lean{Graph.dof_tracking}
  \leanok
  \uses{lem:splitoff-deficiency, lem:removal-deficiency, def:graph-operations}
  Let $v$ be a degree-2 vertex of $G$ with neighbours $a, b$, and let
  $D = \binom{d+1}{2} \ge 2$. If $G$ is a $k$-dof-graph
  ($\operatorname{def}(\tilde G) = k$), then the deficiencies of the
  removal $G_v$ and the splitting-off $G_v^{ab}$ satisfy
  \[
    k - 1 \le \operatorname{def}(\widetilde{G_v^{ab}}) \le k,
    \qquad
    \operatorname{def}(\widetilde{G_v}) \ge k.
  \]
  That is, $G_v^{ab}$ is a $k$-dof- or a $(k-1)$-dof-graph, and $G_v$
  has deficiency at least $k$. These are the dof-conservation laws the
  induction tracks. (Katoh--Tanigawa additionally read off \emph{which}
  alternative holds for $G_v^{ab}$ from the fundamental-circuit count of
  the new edge $ab$, through the forest surgery (\cref{rem:kt-lemma-41});
  that refinement is not needed for Theorem~4.9, which consumes only the
  two-sided bound, so it is omitted.)
\end{lemma}
\begin{proof}
  \leanok
  \uses{lem:splitoff-deficiency, lem:removal-deficiency}
  A packaging of the splitting-off bound
  (\cref{lem:splitoff-deficiency}, both directions) and the removal
  bound (\cref{lem:removal-deficiency}); each is a per-partition
  deficiency-count comparison through the def\,$=$\,corank bridge
  (\cref{thm:def-eq-corank}), no forests. This replaces
  Katoh--Tanigawa's forest-surgery derivation (\cref{rem:kt-lemma-41}).
\end{proof}

\begin{lemma}[Edge-count bound with no proper rigid subgraph; KT Lemma 4.5(i)]
  \label{lem:no-rigid-edge-count}
  \lean{Graph.no_rigid_edge_count}
  \leanok
  \uses{lem:circuit-induces-rigid, def:k-dof, def:rigid-subgraph, def:matroid-MG}
  Let $G$ be a minimal $0$-dof-graph with no proper rigid subgraph and
  $D = \binom{d+1}{2} \ge 2$. Then
  \[
    (D-1)\,|E(G)| \;<\; D\,(|V(G)| - 1) + (D-1).
  \]
  Equivalently, $\operatorname{corank} M(\tilde G) = |E(\tilde G)| -
  \operatorname{rank} M(\tilde G) \le D - 2$: the fibers redundant in
  $M(\tilde G)$ all concentrate on a single edge-fiber. This is the edge bound
  Katoh--Tanigawa use to force a low-degree vertex (\cref{lem:reducible-vertex}).
\end{lemma}
\begin{proof}
  \leanok
  \uses{thm:def-eq-corank}
  Read the graph-level bound through the matroid: $|E(\tilde G)| = (D-1)\,|E(G)|$
  (each edge of $G$ contributes a fiber $\tilde e$ of size $D-1$, the
  \texttt{mulTilde}-level count \texttt{Graph.mulTilde\_edgeSet\_ncard}), and a
  base $B^\ast$ has $|B^\ast| = D(|V(G)| - 1)$ since $G$ is $0$-dof
  (\cref{thm:def-eq-corank}); so the claim is exactly $\operatorname{corank}
  M(\tilde G) \le D - 2$. Fix an edge $e$ and let $h^\ast = \min_B |\tilde e \cap
  B|$ over bases $B$ of $M(\tilde G)$, attained at $B^\ast$; minimality of $G$
  gives $h^\ast \ge 1$. We show $E(\tilde G) \setminus \tilde e \subseteq B^\ast$
  (KT eq.~4.3). If some $f \in E(\tilde G) \setminus \tilde e$ were outside
  $B^\ast$, its fundamental circuit $X = \operatorname{fundCircuit}(f, B^\ast)$
  induces, by \cref{lem:circuit-induces-rigid}, a rigid subgraph $G[V(X)]$; the
  no-proper-rigid hypothesis forces $V(X) = V(G)$ (the spanning step
  \texttt{Graph.fundCircuit\_inducedSpan\_vertexSet\_eq}). Then $X \cap \tilde e
  \neq \emptyset$: otherwise $X \subseteq E(\tilde G) \setminus \tilde e$, and for
  any $e_j \in X$ the set $X - e_j$ is independent and, by
  \cref{lem:circuit-induces-rigid}, of full size $D(|V(X)| - 1) = D(|V(G)| - 1) =
  |B^\ast|$, hence a \emph{base} avoiding $\tilde e$ entirely --- contradicting
  minimality. (This is a direct base-meets-fiber contradiction, not the
  $D$-spanning-trees forest argument of KT; the minimality clause of
  \cref{def:k-dof} carries it.) Picking $e_j \in X \cap \tilde e$, the base
  exchange $B = B^\ast + f - e_j$ (independent by the fundamental-circuit
  characterization, with $f \notin \tilde e$ so $f \neq e_j$) has $|B \cap \tilde
  e| = h^\ast - 1 < h^\ast$, contradicting the choice of $B^\ast$. So
  $E(\tilde G) \setminus \tilde e \subseteq B^\ast$, whence $|E(\tilde G)| =
  |B^\ast| + (|\tilde e| - h^\ast) = D(|V(G)| - 1) + (D - 1 - h^\ast) \le
  D(|V(G)| - 1) + (D - 2)$.
\end{proof}

\begin{lemma}[A low-degree vertex by the average-degree count; KT Lemma 4.6, core]
  \label{lem:low-degree-vertex}
  \lean{Graph.exists_degree_le_two}
  \leanok
  \uses{lem:no-rigid-edge-count, def:k-dof}
  Let $G$ be a minimal $0$-dof-graph with no proper rigid subgraph,
  $V(G) \neq \emptyset$, and $D = \binom{d+1}{2} \ge 3$ (the molecular
  regime $d \ge 2$). Then some vertex $v \in V(G)$ has multigraph degree
  $\deg_G(v) \le 2$.
\end{lemma}
\begin{proof}
  \leanok
  Average-degree count. The edge bound (\cref{lem:no-rigid-edge-count})
  reads $(D-1)\,|E| < D\,(|V|-1) + (D-1) = D|V| - 1$; multiplying by $2$
  and cancelling against $3(D-1)|V|$ (using $D \ge 3$ and $|V| \ge 1$)
  gives the average-degree bound $2|E| < 3|V|$, i.e.\ $d_{\mathrm{avg}} =
  2|E|/|V| < 2D/(D-1) \le 3$. The multigraph handshake $\sum_v \deg_G(v)
  = 2|E|$ then forces some vertex to have degree $< 3$, i.e.\ $\le 2$,
  by a finite pigeonhole. The degree function and the handshake are the
  vendored \texttt{Graph.degree} / \texttt{Graph.handshake\_degree\_subtype}
  API (Peter Nelson's \texttt{apnelson1/Matroid} \texttt{Graph.incFun}
  development), reused here rather than re-derived.
\end{proof}

\begin{lemma}[Existence of a degree-2 vertex; KT Lemma 4.6]
  \label{lem:reducible-vertex}
  \lean{Graph.exists_degree_eq_two}
  \leanok
  \uses{lem:low-degree-vertex, lem:two-edge-conn, def:k-dof}
  A minimal $0$-dof-graph $G$ with no proper rigid subgraph, $|V| \ge 2$,
  and $D = \binom{d+1}{2} \ge 3$ (the molecular regime $d \ge 2$)
  contains a vertex $v$ of multigraph degree exactly $2$. (This degree-2
  vertex is the one whose splitting-off the reduction step
  (\cref{lem:reduction-step}) shows preserves minimality and the degree
  of freedom.)
\end{lemma}
\begin{proof}
  \leanok
  The average-degree count (\cref{lem:low-degree-vertex}) supplies a
  vertex of degree $\le 2$; the $2$-edge-connectivity
  (\cref{lem:two-edge-conn}) rules out degree $\le 1$, pinning it to
  degree exactly $2$. Concretely, the single-vertex cut $\{v\}$ has at
  least two crossing edges (KT~3.1 admits no bridge cut in a $0$-dof
  graph), and the crossing edges of $\{v\}$ are exactly the nonloop
  edges at $v$, so $\deg_G(v) \ge d_G(\{v\}) \ge 2$. The cut count and
  the vendored multigraph degree (\texttt{Graph.degree}) are bridged by
  the observation that an edge crosses $\{v\}$ iff exactly one endpoint
  is $v$. This is the lemma that guarantees the induction can always
  take a step.
\end{proof}

\begin{lemma}[Each reduction strictly shrinks the vertex set; KT Theorem 4.9, measure]
  \label{lem:reduction-measure}
  \lean{Graph.splitOff_vertexSet_ncard_lt, Graph.rigidContract_vertexSet_ncard_lt}
  \leanok
  \uses{def:graph-operations, def:rigid-contraction}
  Both reduction operations strictly decrease the number of vertices, the
  well-founded measure on which Theorem~4.9 inducts. Splitting-off deletes
  the degree-2 vertex, $V(G_v^{ab}) = V(G) \setminus \{v\}$, so
  $|V(G_v^{ab})| < |V(G)|$ for $v \in V(G)$. Contraction collapses $V(H)$ to a
  single representative, so $|V(G / E(H))| \le |V(G)| - |V(H)| + 1 < |V(G)|$
  whenever $|V(H)| \ge 2$ --- the genuine requirement, since contracting a
  single-vertex subgraph is a vertex-set no-op; Case~I always contracts a
  proper rigid subgraph spanning at least two vertices.
\end{lemma}
\begin{proof}
  \leanok
  Splitting-off: immediate from $V(G_v^{ab}) = V(G) \setminus \{v\}$
  (\texttt{Graph.vertexSet\_splitOff}) and $v \in V(G)$. Contraction: the
  image $\operatorname{collapseTo}\,r\,V(H)\,''\,V(G)$ is contained in
  $(V(G) \setminus V(H)) \cup \{r\}$, of cardinality at most $|V(G)| - |V(H)|
  + 1$, which is $< |V(G)|$ exactly when $|V(H)| \ge 2$.
\end{proof}

\begin{lemma}[Reduction step preserves minimality; KT Lemmas 4.7--4.8]
  \label{lem:reduction-step}
  \lean{Graph.splitOff_isMinimalKDof, Graph.contraction_isMinimalKDof}
  \leanok
  \uses{lem:reducible-vertex, lem:contraction-minimality, lem:dof-tracking, lem:reduction-measure, lem:circuit-induces-rigid}
  Each reduction --- splitting-off at the degree-2 vertex of
  \cref{lem:reducible-vertex}, or contraction of a proper rigid
  subgraph (\cref{lem:contraction-minimality}) --- carries a minimal
  $0$-dof-graph (with no proper rigid subgraph, in the splitting-off
  case) to a smaller minimal $0$-dof-graph. This is the inductive step
  of KT Theorem~4.9.
\end{lemma}
\begin{proof}
  \leanok
  \uses{lem:circuit-induces-rigid}
  The contraction branch is \cref{lem:contraction-minimality}. For the
  splitting-off branch (KT 4.8(i)): $\operatorname{def}(\tilde
  G_v^{ab}) = 0$ is \cref{lem:dof-tracking} squeezed against
  $\operatorname{def}(\tilde G) = 0$. For minimality, Katoh--Tanigawa's
  claim (4.10) --- every circuit of $M(\tilde G_v^{ab})$ meets the fresh
  short-circuit fiber $\tilde{ab}$ --- is exactly the statement that the
  surviving ground set $E(\tilde G_v) = E(\tilde G_v^{ab}) \setminus
  \tilde{ab}$ is independent in $M(\tilde G_v^{ab})$: otherwise a circuit
  inside it would be a circuit of $M(\tilde G)$ avoiding $v$, inducing a
  proper rigid subgraph (\cref{lem:circuit-induces-rigid}). So $E(\tilde
  G_v)$ is a base of $M(\tilde G_v)$, and KT 4.7 gives
  $\operatorname{def}(\tilde G_v) > 0$ ($G_v$ is a proper subgraph, so not
  rigid). A \emph{rank count} replaces KT's iterated fundamental-circuit
  swap: any base $B'$ of $M(\tilde G_v^{ab})$ avoiding a fiber $\tilde e$
  has $|B'| \le |E(\tilde G_v)|$ (split $B'$ across $\tilde{ab}$ and
  $E(\tilde G_v)$), which through
  $|B| + \operatorname{def} = D(|V|-1)$ forces $\operatorname{def}(\tilde
  G_v) \le 0$ --- a contradiction. Hence every base meets every fiber.
\end{proof}

\subsection{Theorem 4.9}
\label{sec:molecular-induction-thm49}

\begin{theorem}[Reduction of minimal $0$-dof-graphs; KT Theorem 4.9]
  \label{thm:minimal-kdof-reduction}
  \lean{Graph.minimal_kdof_reduction}
  \leanok
  \uses{lem:reduction-step, lem:reducible-vertex, lem:contraction-minimality, def:graph-operations, lem:reduction-measure}
  Every minimal $0$-dof-graph $G$ with $|V| \ge 2$ reduces to the
  two-vertex double edge by a sequence of splitting-off and
  proper-rigid-subgraph contraction operations. Equivalently, $G$ is
  generated from the two-vertex double edge by the inverse operations
  (edge-splitting and rigid-subgraph expansion). The formalization
  states this as the well-founded induction principle the reduction
  dichotomy and the vertex-count measure (\cref{lem:reduction-measure})
  drive: a property $P$ closed under the two-vertex base case, under
  splitting off a reducible degree-2 vertex, and under contracting a
  proper rigid subgraph (with the induction hypothesis on every
  strictly-smaller minimal $0$-dof-graph) holds of every minimal
  $0$-dof-graph with $|V| \ge 2$.
\end{theorem}
\begin{proof}
  \leanok
  Strong induction on $|V|$. The base case $|V| = 2$ is the two-vertex
  double edge. For $|V| \ge 3$: if $G$ has a proper rigid subgraph, the
  contraction step applies (\cref{lem:contraction-minimality}); otherwise
  \cref{lem:reducible-vertex} supplies a reducible degree-2 vertex, whose
  two incident edges are recovered as the distinct nonloop edges at $v$
  (the $0$-dof two-edge-connectivity rules out a single loop), and
  splitting it off yields a strictly smaller minimal $0$-dof-graph
  (\cref{lem:reduction-step,lem:reduction-measure}) on which the induction
  hypothesis applies. The contraction branch is handed the induction
  hypothesis rather than recursing internally: the matroid-side
  \cref{lem:contraction-minimality} is not yet bridged to a graph-level
  minimality statement for $G / E(H)$ (a single-vertex subgraph is
  vacuously rigid, so the predicate alone need not drop the measure), and
  the splitting-off branch carries an unused-edge-label premise for the
  fresh short-circuit edge. This is
  Katoh--Tanigawa~\cite[Theorem 4.9]{katohTanigawa2011}; it is the
  combinatorial skeleton the algebraic induction (Phases~21--23) realizes
  at the rigidity-matrix rank.
\end{proof}
