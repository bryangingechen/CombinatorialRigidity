\section{The matroid \texorpdfstring{$M(\tilde G)$}{M(G~)}, deficiency, and \texorpdfstring{$k$}{k}-dof graphs}
\label{sec:molecular-deficiency}

This chapter is the \textbf{Phase~19} dep-graph: the combinatorial
substrate of the molecular-conjecture program ---
Katoh--Tanigawa~\cite[\S2.5, \S3]{katohTanigawa2011}, the third
stratum. Where \cref{sec:molecular-rigidity-matrix} built the
\emph{analytic} side (the rigidity matrix $R(G,p)$, its rank, and the
three rank lemmas), this chapter builds the \emph{matroidal} side: the
matroid $M(\tilde G)$, the $D$-deficiency $\operatorname{def}(\tilde
G)$, the $k$-degree-of-freedom hierarchy, rigid subgraphs, and the
bridge $\operatorname{def}(\tilde G) = \operatorname{corank} M(\tilde
G)$ that ties the two sides together. This bridge is the matroidal half
of Tay--Whiteley's Proposition~1.1, reconciling the honest rank form of
\cref{sec:molecular-rigidity-matrix} with the body-hinge existence form
of Phase~16; the remaining analytic half (the generic-rank theorem
$\operatorname{rank} R(G,p) = D(|V|-1) - \operatorname{def}(\tilde G)$)
is presented with the algebraic induction (Phase~21+). Throughout
$D = \binom{d+1}{2} = \delta$ is the body-bar dimension and $\tilde G$
abbreviates the multiplied graph $(D-1)\cdot G$ of
\cref{sec:body-hinge}, whose edge-fibers $\tilde e$ are the $D-1$
parallel copies of each edge $e \in E(G)$.

\subsection{The matroid \texorpdfstring{$M(\tilde G)$}{M(G~)} at the boundary regime}
\label{sec:molecular-deficiency-matroid}

\begin{definition}[The matroid $M(\tilde G)$]
  \label{def:matroid-MG}
  \lean{Graph.matroidMG, Graph.matroidMG_indep_iff}
  \leanok
  \uses{thm:tay-witness}
  Let $D = \binom{d+1}{2}$. The matroid $M(\tilde G)$ is the
  \emph{$(D,D)$-count matroid} of the multiplied graph
  $\tilde G = (D-1)\cdot G$ --- equivalently, the $D$-fold union of the
  cycle (graphic) matroid of $\tilde G$, restricted to $E(\tilde G)$.
  Its independent sets are the $(D,D)$-sparse edge subsets of
  $\tilde G$; by Tutte--Nash-Williams (\cref{thm:tay-witness} and the
  tree-packing chapter) these are exactly the edge subsets that
  decompose into $D$ edge-disjoint forests.
\end{definition}
\begin{proof}
  \leanok
  This is the \textbf{boundary regime} $\ell = 2k = D$ of the
  count-matroid family. It is \emph{not} covered by the matroidal-regime
  construction of \cref{sec:count-matroid} (built for $\ell < 2k$);
  $M(\tilde G)$ is instead the $D$-fold graphic-matroid union of
  \cref{sec:body-bar,sec:body-hinge}, whose independence
  characterization is Tutte--Nash-Williams tree-packing. The
  first formalization step of Phase~19 confirms this boundary regime is
  clean before the rest of the chapter relies on it.
\end{proof}

\begin{definition}[$D$-deficiency]
  \label{def:D-deficiency}
  \lean{Graph.partitionDef, Graph.deficiency}
  \leanok
  \uses{def:matroid-MG}
  For a partition $P$ of $V(G)$ into nonempty parts, write $d_G(P)$ for
  the number of edges of $G$ joining distinct parts of $P$. The
  \emph{$D$-deficiency} of $P$ is
  \[
    \operatorname{def}_{\tilde G}(P) \;=\; D\,(|P| - 1) \;-\;
      (D-1)\,d_G(P),
  \]
  and the \emph{deficiency} of $\tilde G$ is
  $\operatorname{def}(\tilde G) = \max_{P} \operatorname{def}_{\tilde
  G}(P)$, the maximum over all partitions. The deficiency measures how
  far $\tilde G$ is from being independently realizable as a rigid
  body-hinge framework. A partition is encoded by a labeling
  $f \colon V \to V$ whose fibers are the parts; $|P|$ is the number of
  distinct labels on $V(G)$ and $d_G(P)$ the number of edges with
  differently-labeled endpoints. The quantity
  $\operatorname{def}_{\tilde G}(P)$ is genuinely signed (a fine
  partition crossing many edges is negative), so it is $\mathbb Z$-valued;
  the maximum is $\ge 0$, witnessed by the trivial one-part partition.
\end{definition}

\begin{definition}[$k$-dof and minimal $k$-dof graphs]
  \label{def:k-dof}
  \lean{Graph.IsKDof, Graph.IsMinimalKDof, Graph.edgeFiber}
  \leanok
  \uses{def:D-deficiency, def:matroid-MG}
  A multigraph $G$ is a \emph{$k$-dof-graph} when
  $\operatorname{def}(\tilde G) = k$; the $0$-dof case is exactly the
  body-hinge-rigid case (\cref{thm:body-hinge-tay}). $G$ is a
  \emph{minimal $k$-dof-graph} when it is $k$-dof and every base $B$ of
  $M(\tilde G)$ meets every edge-fiber $\tilde e$ (the $D-1$ parallel
  copies of $e \in E(G)$) --- i.e.\ no edge of $G$ can be deleted
  without changing the deficiency. Minimal $k$-dof-graphs are the
  objects the combinatorial induction (\cref{sec:molecular-deficiency},
  Phase~20's Theorem~4.9) reduces to the two-vertex double edge.
\end{definition}

\subsection{Rigid subgraphs and structural lemmas}
\label{sec:molecular-deficiency-rigid}

\begin{definition}[Rigid and proper rigid subgraphs]
  \label{def:rigid-subgraph}
  \lean{Graph.IsRigidSubgraph, Graph.IsProperRigidSubgraph}
  \leanok
  \uses{def:k-dof}
  A subgraph $H \subseteq G$ (the multigraph subgraph order $H \le G$)
  is \emph{rigid} when it is $0$-dof ($\operatorname{def}(\tilde H) =
  0$), i.e.\ $\tilde H$ packs $D$ edge-disjoint spanning trees; it is a
  \emph{proper} rigid subgraph when additionally
  $\emptyset \ne V(H) \subsetneq V(G)$. A \emph{circuit} of
  $M(\tilde G)$ is a minimal dependent edge set (mathlib's
  \texttt{Matroid.IsCircuit}). Two-edge-connectivity of $G$ is the
  structural hypothesis Case~III of the algebraic induction works
  under (\cref{lem:two-edge-conn}).
\end{definition}

\begin{lemma}[Two-edge-connectivity, cut form; KT Lemma 3.1]
  \label{lem:two-edge-conn}
  \lean{Graph.two_le_crossingEdges_of_isKDof_zero, Graph.cutLabeling, Graph.numParts_cutLabeling}
  \leanok
  \uses{def:k-dof, def:D-deficiency}
  A body-hinge-rigid ($0$-dof) graph $G$ is $2$-edge-connected: for any
  nonempty proper vertex set $V' \subsetneq V(G)$ separating $V(G)$, at
  least two edges cross the cut $\{V', V \setminus V'\}$, i.e.\
  $d_G(V') \ge 2$.
\end{lemma}
\begin{proof}
  \leanok
  Were $d_G(V') \le 1$, the two-part cut partition $P = \{V', V
  \setminus V'\}$ would witness $\operatorname{def}(\tilde G) \ge
  \operatorname{def}_{\tilde G}(P) = D(|P| - 1) - (D-1)\,d_G(P) \ge D -
  (D-1) = 1 > 0$, contradicting $\operatorname{def}(\tilde G) = 0$
  (the deficiency is an upper bound for every partition's deficiency).
  This is Katoh--Tanigawa~\cite[Lemma 3.1]{katohTanigawa2011}. As
  mathlib carries no edge-connectivity predicate for multigraphs, the
  conclusion is phrased directly in the cut form the proof uses; the
  cut is encoded by the labeling $f$ collapsing $V'$ and its complement
  to two distinct representatives ($D = \operatorname{bodyBarDim} n \ge
  1$).
\end{proof}

\begin{lemma}[Matroid restriction descends to subgraphs]
  \label{lem:matroid-restrict-subgraph}
  \lean{Graph.matroidMG_restrict_mulTilde}
  \leanok
  \uses{def:matroid-MG}
  For a subgraph $H \le G$, restricting $M(\tilde G)$ to the edge set
  $E(\tilde H)$ of the smaller multiplied graph recovers $M(\tilde H)$:
  \[
    M(\tilde G)\,|\,E(\tilde H) \;=\; M(\tilde H).
  \]
\end{lemma}
\begin{proof}
  \leanok
  Matroid restriction commutes with the $D$-fold cycle-matroid union
  construction. Both sides are checked equal on independent sets: an
  edge set $E' \subseteq E(\tilde H)$ is independent in either matroid
  iff it is $(D,D)$-sparse, and the two sparsity conditions agree
  because the incidences of an edge of $\tilde H$ coincide with its
  incidences in $\tilde G$ (the subgraph relation cannot be seen by a
  restriction to a fiber subset of the smaller graph). This is the
  identity Katoh--Tanigawa~\cite[Lemma 3.3]{katohTanigawa2011} runs on.
\end{proof}

\begin{lemma}[Subgraph minimality via restriction; KT Lemma 3.3]
  \label{lem:subgraph-minimality}
  \lean{Graph.subgraph_minimality}
  \leanok
  \uses{def:k-dof, lem:matroid-restrict-subgraph}
  Let $G$ be a minimal $k$-dof-graph and let $H \le G$ be a subgraph
  with $\operatorname{def}(\tilde H) = k'$. Then $H$ is a minimal
  $k'$-dof-graph. In particular a rigid subgraph ($k' = 0$) of a
  minimal $k$-dof-graph is a minimal $0$-dof-graph.
\end{lemma}
\begin{proof}
  \leanok
  The $\operatorname{def}(\tilde H) = k'$ half is the choice of $k'$;
  the content is the base / fiber-meeting minimality transport. By
  \cref{lem:matroid-restrict-subgraph}, $M(\tilde H) = M(\tilde
  G)\,|\,E(\tilde H)$, so any base $B'$ of $M(\tilde H)$ is an
  $M(\tilde G)$-basis of $E(\tilde H)$ and extends to a base $B$ of
  $M(\tilde G)$ with $B' = B \cap E(\tilde H)$ by maximality. Each
  edge-fiber $\tilde e$ of an edge $e \in E(H) \subseteq E(G)$ lies
  inside $E(\tilde H)$, so $G$'s minimality ($B \cap \tilde e \ne
  \emptyset$) descends: $B' \cap \tilde e = (B \cap E(\tilde H)) \cap
  \tilde e = B \cap \tilde e \ne \emptyset$. This is
  Katoh--Tanigawa~\cite[Lemma 3.3]{katohTanigawa2011}.
\end{proof}

\begin{lemma}[A circuit minus an edge is a maximal sparse subset; KT Lemma 3.4]
  \label{lem:circuit-rigid}
  \lean{Graph.isSparse_diff_singleton_of_isCircuit}
  \leanok
  \uses{def:rigid-subgraph, def:matroid-MG}
  Let $X$ be a circuit of $M(\tilde G)$ and let $e \in X$. Then
  $X - e$ is $(D,D)$-sparse --- equivalently, $X - e$ is an
  $M(\tilde G)$-basis of $X$: a circuit is exactly one edge short of
  being independent on its vertex span.
\end{lemma}
\begin{proof}
  \leanok
  $X - e$ is independent since $X$ is a minimal dependent set, so by
  boundary-regime cleanliness (\cref{def:matroid-MG}) it is
  $(D,D)$-sparse; and $X$ is dependent, so $X - e$ is a maximal
  independent --- i.e.\ basis --- subset of $X$. This is the
  upper-bound / maximal-sparse-subset half of
  Katoh--Tanigawa~\cite[Lemma 3.4]{katohTanigawa2011}. KT's full
  conclusion --- that the vertex-induced subgraph $G[V(X)]$ is rigid,
  more precisely that $X - e$ partitions into $D$ edge-disjoint
  spanning trees on $V(X)$, i.e.\ $|X - e| = D(|V(X)| - 1)$ exactly ---
  needs the lower bound $|X| > D(|V(X)| - 1)$ forced by $X$ being
  dependent, which in this count matroid is the reverse direction of
  the $\operatorname{def} = \operatorname{corank}$ min--max
  (\cref{thm:def-eq-corank}); it is deferred with that node.
\end{proof}

\subsection{The deficiency \texorpdfstring{$=$}{=} corank bridge}
\label{sec:molecular-deficiency-bridge}

\begin{lemma}[Rank upper bound for $M(\tilde G)$]
  \label{lem:rank-matroidMG-le}
  \lean{Graph.rank_matroidMG_le}
  \leanok
  \uses{def:matroid-MG}
  For a multigraph $G$ with $V(G) \ne \emptyset$,
  \[
    \operatorname{rank} M(\tilde G) \;\le\; D\,(|V(G)| - 1).
  \]
\end{lemma}
\begin{proof}
  \leanok
  Every base $B$ of $M(\tilde G)$ is independent, hence (boundary-regime
  cleanliness, \cref{def:matroid-MG}) the bar set $B$ is $(D,D)$-sparse
  in $\tilde G$. Applying sparsity to $B$ itself gives $|B| + D \le D
  \cdot |V'|$, where $V' \subseteq V(\tilde G) = V(G)$ are the vertices
  $B$ spans; since $|V'| \le |V(G)|$ and $|V(G)| \ge 1$, this rearranges
  to $|B| \le D\,(|V(G)| - 1)$. This is the matroidal upper-bound half of
  \cref{thm:def-eq-corank} (the codimension fact $\operatorname{rank} R
  \le D(|V|-1)$ on the analytic side,
  \cref{sec:molecular-rigidity-matrix}).
\end{proof}

\begin{lemma}[Partition-respecting cycle-matroid rank bound]
  \label{lem:rk-within-parts}
  \lean{Graph.rk_cycleMatroid_within_parts_le}
  \leanok
  \uses{def:matroid-MG, def:D-deficiency}
  Let $P$ be a partition of $V(G)$ and let $Y \subseteq E(\tilde G)$ be the set
  of fibers of $\tilde G$ whose two endpoints lie in the same part of $P$
  (the non-crossing fibers). Then the cycle-matroid rank of $Y$ satisfies
  \[
    \operatorname{r}_{\mathrm{cycle}}(Y) + |P| \;\le\; |V(G)|,
  \]
  i.e.\ $\operatorname{r}_{\mathrm{cycle}}(Y) \le |V(G)| - |P|$.
\end{lemma}
\begin{proof}
  \leanok
  By the cycle-matroid rank--component identity
  $\operatorname{r}_{\mathrm{cycle}}(Y) + c(\tilde G \restriction Y) = |V(G)|$
  (the edge-restriction $\tilde G \restriction Y$ keeps every vertex, so
  $V(\tilde G \restriction Y) = V(\tilde G) = V(G)$). It remains to see
  $c(\tilde G \restriction Y) \ge |P|$. A fiber of $Y$ joins two
  equally-labelled vertices, so along any walk of $\tilde G \restriction Y$ the
  part label is constant; hence the part containing a vertex is a connected-
  component invariant, and the map sending each part to the component of one of
  its vertices is injective. Thus the number of components is at least the
  number $|P|$ of parts.
\end{proof}

\begin{lemma}[Weak duality: $\operatorname{def}(\tilde G) \le \operatorname{corank} M(\tilde G)$]
  \label{lem:weak-duality}
  \lean{Graph.rank_add_partitionDef_le, Graph.rank_add_deficiency_le}
  \leanok
  \uses{def:D-deficiency, def:matroid-MG, lem:rk-within-parts}
  For a multigraph $G$ with $V(G) \ne \emptyset$ and $D \ge 1$, and every
  partition $P$ of $V(G)$,
  \[
    \operatorname{rank} M(\tilde G) + \operatorname{def}_{\tilde G}(P)
      \;\le\; D\,(|V(G)| - 1);
  \]
  maximising over $P$ gives
  $\operatorname{rank} M(\tilde G) + \operatorname{def}(\tilde G) \le
  D\,(|V(G)| - 1)$, i.e.\ $\operatorname{def}(\tilde G) \le
  \operatorname{corank} M(\tilde G)$.
\end{lemma}
\begin{proof}
  \leanok
  Fix $P$, encoded by $f$, and let $Y \subseteq E(\tilde G)$ be the
  non-crossing fibers (those whose underlying edge does not cross $P$). The
  Edmonds matroid-partition rank formula (\cref{sec:matroid-union},
  \texttt{Union\_pow\_rk\_eq}) applied to the $D$-fold cycle-matroid union on
  $X = E(\tilde G)$ with the witness $Y$ gives
  $\operatorname{rank} M(\tilde G) \le D \cdot
  \operatorname{r}_{\mathrm{cycle}}(Y) + |E(\tilde G) \setminus Y|$. By
  \cref{lem:rk-within-parts}, $D \cdot \operatorname{r}_{\mathrm{cycle}}(Y)
  \le D(|V| - |P|)$; and the crossing fibers $E(\tilde G) \setminus Y$ number
  exactly $(D-1)\,d_G(P)$ (each of the $d_G(P)$ crossing edges of $G$
  contributes its $D-1$ parallel copies). Substituting,
  $\operatorname{rank} M(\tilde G) + \operatorname{def}_{\tilde G}(P) =
  \operatorname{rank} M(\tilde G) + D(|P|-1) - (D-1)\,d_G(P) \le D(|V| - 1)$.
  Maximising over $P$ (a conditionally-complete supremum, bounded by the
  per-partition bound) gives the deficiency form.
\end{proof}

\begin{theorem}[$\operatorname{def}(\tilde G) = \operatorname{corank} M(\tilde G)$; Jackson--Jord\'an]
  \label{thm:def-eq-corank}
  \lean{Graph.rank_add_deficiency_eq, Graph.isBase_ncard_add_deficiency_eq, Graph.le_rank_add_deficiency}
  \leanok
  \uses{def:D-deficiency, def:matroid-MG, lem:rank-matroidMG-le, lem:rk-within-parts, lem:weak-duality}
  For any base $B$ of $M(\tilde G)$,
  \[
    |B| + \operatorname{def}(\tilde G) \;=\; D\,(|V| - 1),
  \]
  i.e.\ $\operatorname{def}(\tilde G) = \operatorname{corank} M(\tilde
  G)$ inside the rank-$D(|V|-1)$ ambient. This is the project framing of
  the generic-rank identity of Jackson--Jord\'an~\cite[Theorem
  6.1]{jacksonJordan2009}, whose Corollary~6.2 reads off rigidity
  ($\operatorname{def} = 0$) as $\tilde G$ packing $D$ edge-disjoint
  spanning trees.
\end{theorem}
\begin{proof}
  \leanok
  The rank of the $D$-fold graphic-matroid union is the partition
  formula of Edmonds (\cref{sec:matroid-union}); the corank against the
  ambient $D(|V|-1)$ is exactly the deficiency maximised over
  partitions. One inequality is weak duality (\cref{lem:weak-duality}):
  $\operatorname{def}(\tilde G) \le \operatorname{corank} M(\tilde G)$,
  from the Edmonds rank formula and the partition-respecting bound
  \cref{lem:rk-within-parts}, alongside the rank upper bound
  \cref{lem:rank-matroidMG-le} (matching, on the matroidal side, the
  body-hinge reduction of \cref{thm:body-hinge-tay}). The reverse
  inequality $\operatorname{def}(\tilde G) \ge \operatorname{corank}
  M(\tilde G)$ --- the substantive Jackson--Jord\'an~\cite{jacksonJordan2009}
  min--max content --- exhibits a vertex-partition attaining the rank: the
  existence half of the Edmonds formula provides an optimal edge set
  $Y_0 \subseteq E(\tilde G)$ with $D\cdot
  \operatorname{r}_{\mathrm{cycle}}(Y_0) + |E(\tilde G) \setminus Y_0| \le
  \operatorname{rank} M(\tilde G)$, and the partition $P$ into the connected
  components of $\tilde G \restriction Y_0$ is then attaining. For that $P$
  every fiber of $Y_0$ joins same-component (so equally-labelled) vertices,
  whence $\operatorname{r}_{\mathrm{cycle}}(Y_0) + |P| = |V|$ exactly (the
  $\le$ is \cref{lem:rk-within-parts}, the $\ge$ the rank--component
  identity together with the components-refine-the-partition map run in
  reverse); and every crossing edge has all $D-1$ of its fibers outside
  $Y_0$, so $(D-1)\,d_G(P) \le |E(\tilde G) \setminus Y_0|$. Substituting,
  $\operatorname{rank} M(\tilde G) \ge D(|V| - |P|) + (D-1)\,d_G(P) =
  D(|V|-1) - \operatorname{def}_{\tilde G}(P)$, so
  $\operatorname{def}(\tilde G) \ge \operatorname{corank} M(\tilde G)$. The
  two inequalities give the equality.
\end{proof}

This bridge is the matroidal half of Tay--Whiteley's
Proposition~1.1, which equates the honest rank form
$\operatorname{rank} R(G,p) = D(|V|-1)$ of
\cref{sec:molecular-rigidity-matrix} with the standard-basis existence
form (\cref{thm:body-hinge-tay}) of Phase~16. The reconciliation routes
through $\operatorname{def}(\tilde G) = \operatorname{corank} M(\tilde
G)$ (\cref{thm:def-eq-corank}, now green); the remaining \emph{analytic}
link --- the generic-rank theorem $\operatorname{rank} R(G,p) = D(|V|-1)
- \operatorname{def}(\tilde G)$ (Jackson--Jord\'an Thm 6.1, geometric
side) --- wires the rigidity-matrix rank to the matroid corank and is
presented with the algebraic induction (Phase~21+).
