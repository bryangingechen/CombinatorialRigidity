\section{The \texorpdfstring{$(k, \ell)$}{(k, l)}-count matroid}
\label{sec:count-matroid}

\Cref{sec:rigidity-matroid} characterises planar row-independence by
$(2, 3)$-sparsity. Before packaging that result as a matroid in
\cref{sec:rigidity-matroid-instance}, we record the abstract object it
realises: the \emph{$(k, \ell)$-count matroid} on the edge set of
$K_{V}$, defined for finite $V$ whenever $\ell < 2k$. This matroid has
no linear-algebra content; it is determined entirely by the
combinatorial $(k, \ell)$-sparsity condition of \cref{def:isSparse}.
Its independent sets are exactly the $(k, \ell)$-sparse edge sets, and
the planar rigidity matroid is its $(k, \ell) = (2, 3)$ specialisation,
identified with the linear matroid of the rigidity matrix via
\cref{thm:edgeSet-rowIndependent-iff-isSparse}.

This is the standard $(k, \ell)$-count matroid construction, introduced
by Whiteley~\cite{whiteley1996} and developed in the ``pebble game''
framework of Lee--Streinu~\cite{leeStreinu2008}. The range $\ell < 2k$
is the maximal regime in which the $(k, \ell)$-sparse edge sets are
exactly the independent sets of a matroid on $E(K_{V})$; outside it the
bases need not have equal cardinality (for instance, $(3, 6)$-tight
graphs in three-dimensional rigidity).

\emph{Terminology.} For an edge set $I \subseteq E(K_{V})$, call a
subset $S \subseteq V$ that is $(k, \ell)$-tight in
$\mathrm{fromEdgeSet}\,I$ (\cref{def:isTightOn}) \emph{$I$-tight}.
Lee--Streinu and Whiteley call an $I$-tight subset an \emph{$I$-block}
(or \emph{$(k, \ell)$-block} of $I$), and
Jord\'an~\cite{jordan2016} calls it a \emph{critical set} in the
$(2, 3)$ regime (already cross-referenced in
\cref{sec:rigidity-matroid}'s Terminology aside). The maximal $I$-block
containing a fixed finite subset $X \subseteq V$ --- our
$\mathrm{maxBlock}\,X$ (\cref{def:isSparse-maxBlock}) --- is also
known, when $|X| \ge 2$, as the \emph{$(k, \ell)$-tight component} or
\emph{$I$-component} of $X$. The augmentation argument uses this object
only at two-vertex anchors $X = \{u, v\}$, but the construction is
uniform in $X$ once $|X| \ge 2$. These vertex-subset blocks and
components should not be confused with Jord\'an's \emph{$M$-circuit}
(a minimal dependent edge set of the matroid, e.g.\ $K_{4}$ or
$K_{3, 3} + e$) or \emph{$M$-connected component} (matroid
connectivity at the edge level, via shared circuits); those are
matroid-side objects of a different kind, and this chapter does not
use them.

\subsection{Augmentation in the matroidal regime}
\label{sec:count-matroid-aug}

This subsection proves the augmentation axiom of the count matroid:
for $(k, \ell)$-sparse edge sets $I, J$ with $|I| < |J|$, some edge of
$J$ can be added to $I$ keeping it sparse (\cref{lem:isSparse-aug}).
The proof follows Lee--Streinu~\cite{leeStreinu2008} (and Whiteley's
appendix to~\cite{whiteley1996}), through the $I$-component
decomposition of a sparse graph. In the matroidal regime $\ell < 2k$,
two $I$-tight subsets that share at least two vertices have $I$-tight
union and intersection (\cref{lem:isTightOn-union-inter-of-pair}); so
for any finite $X \subseteq V$ with $|X| \ge 2$, the $I$-tight subsets
containing $X$ are closed under union, and their union --- the
$I$-component of $X$ --- is the largest among them. Distinct
$I$-components share at most one vertex, hence no edge, so the
$I$-components of a graph are pairwise edge-disjoint.

The preliminary lemmas below establish this $I$-component structure, in
the order the augmentation argument uses it: the matroidal-regime
closure of $I$-tight subsets under union and intersection, the
extraction lemma turning a sparsity violation into a witnessing
$I$-block, the maximal $I$-block (the $I$-component) itself, and
edge-disjointness of distinct $I$-components.

\begin{lemma}[Block closure under a shared pair, matroidal regime]
  \label{lem:isTightOn-union-inter-of-pair}
  \lean{SimpleGraph.IsTightOn.union_inter_of_pair}
  \leanok
  \uses{def:isTightOn, thm:isTightOn-union-inter}
  Let $V$ be finite and $G$ a $(k, \ell)$-sparse graph on $V$ with
  $\ell < 2k$. Let $s, t \subseteq V$ be two $(k, \ell)$-tight subsets
  of $G$ sharing two distinct vertices $u \ne v$ (that is,
  $u, v \in s \cap t$). Then both $s \cup t$ and $s \cap t$ are
  $(k, \ell)$-tight.
\end{lemma}
\begin{proof}
  \leanok
  \uses{thm:isTightOn-union-inter}
  Apply \cref{thm:isTightOn-union-inter}, whose size proviso
  $\ell \le k\,|s \cap t|$ holds because
  $|s \cap t| \ge |\{u, v\}| = 2$ and $\ell < 2k = k \cdot 2$.
\end{proof}

\begin{lemma}[$I$-block extraction from a sparsity violation]
  \label{lem:exists-isTightOn-of-insert-not-sparse}
  \lean{SimpleGraph.IsSparse.exists_isTightOn_of_insert_not_sparse}
  \leanok
  \uses{def:isSparse, def:isTightOn}
  Let $V$ be finite and $I \subseteq \mathrm{Sym}_{2} V$ with
  $\mathrm{fromEdgeSet}\,I$ $(k, \ell)$-sparse. Let $u \ne v$ be
  vertices whose pair $e = \{u, v\}$ satisfies $e \notin I$, and
  suppose $\mathrm{fromEdgeSet}\,(I \cup \{e\})$ is not
  $(k, \ell)$-sparse. Then some finite subset $S \subseteq V$
  containing both $u$ and $v$ is $(k, \ell)$-tight in
  $\mathrm{fromEdgeSet}\,I$.
\end{lemma}
\begin{proof}
  \leanok
  \uses{def:isSparse, def:isTightOn}
  The sparsity violation yields a subset $S \subseteq V$ with
  $\ell \le k|S|$ and
  $|\edgesIn[\mathrm{fromEdgeSet}\,(I \cup \{e\})]{S}| + \ell > k|S|$.
  Since $e = \{u, v\}$ with $u \ne v$, adding $e$ to $I$ leaves
  $|\edgesIn[\mathrm{fromEdgeSet}\,I]{S}|$ unchanged when
  $\{u, v\} \not\subseteq S$, and increases it by exactly one when
  $\{u, v\} \subseteq S$ (here $e \notin I$ is used). In the first case
  the violation reads
  $|\edgesIn[\mathrm{fromEdgeSet}\,I]{S}| + \ell > k|S|$, contradicting
  sparsity of $\mathrm{fromEdgeSet}\,I$ at $S$; in the second it forces
  $|\edgesIn[\mathrm{fromEdgeSet}\,I]{S}| + \ell = k|S|$, so $S$ is
  $(k, \ell)$-tight in $\mathrm{fromEdgeSet}\,I$.
\end{proof}

\begin{definition}[Maximal $I$-block containing a finite subset]
  \label{def:isSparse-maxBlock}
  \lean{SimpleGraph.maxBlock}
  \leanok
  \uses{def:isSparse, def:isTightOn}
  Let $V$ be finite, $I \subseteq \mathrm{Sym}_{2}V$, and $X \subseteq V$
  a finite subset. The \emph{maximal $I$-block containing $X$}, written
  $\mathrm{maxBlock}\,X$, is the union of all $I$-tight subsets $S$ with
  $X \subseteq S$; it is empty when no such $S$ exists.
\end{definition}

The next lemma shows that in the matroidal regime this union is again
$I$-tight, so $\mathrm{maxBlock}\,X$ is the largest $I$-tight subset
containing $X$ --- the $I$-component of $X$.

\begin{lemma}[$\mathrm{maxBlock}$ is $I$-tight in the matroidal regime]
  \label{lem:isSparse-maxBlock-isTightOn}
  \lean{SimpleGraph.IsSparse.maxBlock_isTightOn}
  \leanok
  \uses{def:isSparse-maxBlock, lem:isTightOn-union-inter-of-pair}
  Let $V$ be finite, $\mathrm{fromEdgeSet}\,I$ be $(k, \ell)$-sparse
  with $\ell < 2k$, and $X \subseteq V$ a finite subset with
  $|X| \ge 2$. If some $I$-tight subset contains $X$, then
  $\mathrm{maxBlock}\,X$ is $I$-tight.
\end{lemma}
\begin{proof}
  \leanok
  \uses{lem:isTightOn-union-inter-of-pair}
  Pick distinct $u, v \in X$ (possible since $|X| \ge 2$). Every
  $I$-tight subset containing $X$ contains the pair $\{u, v\}$, so by
  \cref{lem:isTightOn-union-inter-of-pair} the family $\mathcal{F}$ of
  such subsets is closed under pairwise union. Hence its union
  $\mathrm{maxBlock}\,X = \bigcup \mathcal{F}$ lies in $\mathcal{F}$
  (by induction on $|\mathcal{F}|$: one element lies in $\mathcal{F}$
  trivially, and adjoining a further $T \in \mathcal{F}$ to a union
  $T' \in \mathcal{F}$ gives $T \cup T' \in \mathcal{F}$), and is
  therefore $I$-tight.
\end{proof}

\begin{lemma}[Edge-disjointness of distinct $I$-components]
  \label{lem:isSparse-maxBlock-edge-disjoint}
  \lean{SimpleGraph.IsSparse.maxBlock_eq_of_subset_maxBlock}
  \leanok
  \uses{def:isSparse-maxBlock, lem:isSparse-maxBlock-isTightOn,
        lem:isTightOn-union-inter-of-pair}
  Let $V$ be finite, $\mathrm{fromEdgeSet}\,I$ be $(k, \ell)$-sparse
  with $\ell < 2k$, and $X, Y \subseteq V$ finite subsets with
  $|X|, |Y| \ge 2$. If some $I$-tight subset contains $X$ and
  $Y \subseteq \mathrm{maxBlock}\,X$, then
  $\mathrm{maxBlock}\,Y = \mathrm{maxBlock}\,X$. In particular,
  distinct non-empty $I$-components are edge-disjoint: two of them share
  at most one vertex, hence no off-diagonal edge of $K_{V}$.
\end{lemma}
\begin{proof}
  \leanok
  \uses{lem:isTightOn-union-inter-of-pair,
        lem:isSparse-maxBlock-isTightOn}
  By \cref{lem:isSparse-maxBlock-isTightOn}, $\mathrm{maxBlock}\,X$ is
  an $I$-tight subset containing $Y$, so it is one of the sets whose
  union defines $\mathrm{maxBlock}\,Y$; hence
  $\mathrm{maxBlock}\,X \subseteq \mathrm{maxBlock}\,Y$. Conversely,
  $\mathrm{maxBlock}\,Y$ is $I$-tight
  (\cref{lem:isSparse-maxBlock-isTightOn} at $Y$, whose hypothesis holds
  because the non-empty $I$-tight $\mathrm{maxBlock}\,X$ contains $Y$)
  and shares the two-or-more vertices of $Y$ with $\mathrm{maxBlock}\,X$;
  by \cref{lem:isTightOn-union-inter-of-pair} their union is $I$-tight
  and contains $X$, and maximality of $\mathrm{maxBlock}\,X$ forces
  $\mathrm{maxBlock}\,Y \subseteq \mathrm{maxBlock}\,X$.
\end{proof}

\begin{lemma}[$(k, \ell)$-sparsity augmentation, matroidal regime]
  \label{lem:isSparse-aug}
  \lean{SimpleGraph.IsSparse.exists_aug_of_lt_two_mul}
  \leanok
  \uses{def:isSparse, def:isTightOn,
        lem:isTightOn-union-inter-of-pair,
        lem:exists-isTightOn-of-insert-not-sparse,
        def:isSparse-maxBlock,
        lem:isSparse-maxBlock-isTightOn,
        lem:isSparse-maxBlock-edge-disjoint}
  Let $V$ be finite, $k, \ell \in \mathbb{N}$ with $\ell < 2k$, and let
  $I, J \subseteq E(K_{V})$ be edge sets with $\mathrm{fromEdgeSet}\,I$
  and $\mathrm{fromEdgeSet}\,J$ both $(k, \ell)$-sparse and
  $|I| < |J|$. Then there exists $e \in J \setminus I$ such that
  $\mathrm{fromEdgeSet}\,(I \cup \{e\})$ is $(k, \ell)$-sparse.
\end{lemma}
\begin{proof}
  \leanok
  \uses{def:isSparse, def:isTightOn,
        lem:isTightOn-union-inter-of-pair,
        lem:exists-isTightOn-of-insert-not-sparse,
        def:isSparse-maxBlock,
        lem:isSparse-maxBlock-isTightOn,
        lem:isSparse-maxBlock-edge-disjoint}
  Suppose for contradiction that $\mathrm{fromEdgeSet}\,(I \cup \{e\})$
  is not $(k, \ell)$-sparse for every $e = uv \in J \setminus I$. By
  \cref{lem:exists-isTightOn-of-insert-not-sparse}, each such $e$ then
  lies in an $I$-tight subset containing both endpoints $u$ and $v$.

  For each $e = uv \in J \setminus I$, set
  $C_{e} := \mathrm{maxBlock}\,\{u, v\}$
  (\cref{def:isSparse-maxBlock}). By
  \cref{lem:isSparse-maxBlock-isTightOn} each $C_{e}$ is non-empty and
  $I$-tight --- the $I$-component of $e$. By
  \cref{lem:isSparse-maxBlock-edge-disjoint} distinct $I$-components
  share at most one vertex, so each off-diagonal edge of $K_{V}$ lies
  in at most one of them; the $I$-components are therefore pairwise
  edge-disjoint.

  Now count. Let $\mathrm{Comps}$ be the family of $I$-components, and
  call an edge \emph{free} if it has an endpoint lying in no
  $I$-component; write $I_{\mathrm{free}} \subseteq I$ and
  $J_{\mathrm{free}} \subseteq J$ for the free edges of each. Every
  $e \in J \setminus I$ has both endpoints in $C_{e}$, so no free edge
  lies in $J \setminus I$; hence $J_{\mathrm{free}} \subseteq I$, and in
  fact $J_{\mathrm{free}} \subseteq I_{\mathrm{free}}$, giving
  $|J_{\mathrm{free}}| \le |I_{\mathrm{free}}|$. At each
  $C \in \mathrm{Comps}$, sparsity of $\mathrm{fromEdgeSet}\,J$ at $C$
  and tightness of $\mathrm{fromEdgeSet}\,I$ at $C$ give
  $|\edgesIn[\mathrm{fromEdgeSet}\,J]{C}| \le k|C| - \ell =
  |\edgesIn[\mathrm{fromEdgeSet}\,I]{C}|$. Summing over the
  edge-disjoint $I$-components and adding the free edges,
  \[
    |J| = \sum_{C \in \mathrm{Comps}}
    |\edgesIn[\mathrm{fromEdgeSet}\,J]{C}| + |J_{\mathrm{free}}|
    \le \sum_{C \in \mathrm{Comps}}
    |\edgesIn[\mathrm{fromEdgeSet}\,I]{C}| + |I_{\mathrm{free}}|
    = |I|,
  \]
  contradicting $|I| < |J|$.

  In the upper range $\ell \in [k, 2k)$ every edge lies in a component,
  so the free-edge terms vanish (\cite[Theorem 5(4)]{leeStreinu2008});
  the argument above handles both ranges uniformly.
\end{proof}

The $I$-component argument above establishes the augmentation axiom,
and hence the matroid, without reference to any algorithm.
Lee--Streinu~\cite{leeStreinu2008} also decide sparsity
algorithmically, through the \emph{$(k, \ell)$-pebble game}: a greedy
procedure that maintains $k$ pebbles per vertex and a partial
orientation of the accepted edges, admitting a candidate edge
$\{u, v\}$ exactly when $\mathrm{peb}(u) + \mathrm{peb}(v) \ge \ell + 1$.
\Cref{sec:pebble-game} formalises this decision procedure and proves it
correct in the matroidal regime $\ell < 2k$.

\subsection{The matroid}
\label{sec:count-matroid-def}

Packaging the augmentation as a matroid uses mathlib's
\texttt{IndepMatroid.ofFinite}: the empty edge set is sparse, any
subset of a sparse edge set is sparse, and every sparse edge set is
contained in the ground set $E(K_{V})$ --- the three routine
independence axioms --- while the augmentation axiom is
\cref{lem:isSparse-aug}.

\begin{definition}[$(k, \ell)$-count matroid]
  \label{def:countMatroid}
  \lean{SimpleGraph.countMatroid}
  \leanok
  \uses{def:isSparse, lem:isSparse-aug}
  For finite $V$ and $k, \ell \in \mathbb{N}$ with $\ell < 2k$, the
  \emph{$(k, \ell)$-count matroid on $V$} is the matroid on the ground
  set $E(K_{V})$ (the off-diagonal symmetric pairs over $V$) whose
  independent sets are exactly the $(k, \ell)$-sparse edge sets:
  \[
    \mathrm{Indep}(I) \;\Longleftrightarrow\; I \subseteq E(K_{V})
    \ \text{and}\ \mathrm{fromEdgeSet}\,I \ \text{is}\
    (k, \ell)\text{-sparse}.
  \]
\end{definition}

\begin{theorem}[Independent sets of the count matroid]
  \label{thm:countMatroid-indep-iff}
  \lean{SimpleGraph.countMatroid_indep_iff}
  \leanok
  \uses{def:countMatroid, def:isSparse}
  For finite $V$, $k, \ell \in \mathbb{N}$ with $\ell < 2k$, and
  $I \subseteq \mathrm{Sym}_{2}V$: the edge set $I$ is independent in
  the $(k, \ell)$-count matroid on $V$ if and only if
  $I \subseteq E(K_{V})$ and $\mathrm{fromEdgeSet}\,I$ is
  $(k, \ell)$-sparse.
\end{theorem}
\begin{proof}
  \leanok
  \uses{def:countMatroid}
  Immediate from \cref{def:countMatroid}: the count matroid's
  independent sets are defined to be exactly these edge sets.
\end{proof}
