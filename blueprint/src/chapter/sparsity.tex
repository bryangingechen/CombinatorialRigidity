\section{Edges in a vertex set}
\label{sec:edgesIn}

Throughout, let $G$ be a simple graph on a vertex type $V$ and let
$S, T \subseteq V$. Almost every combinatorial argument in this
project is driven by a single selector --- the set of edges of $G$
whose endpoints both lie in $S$ --- so we begin by collecting its
basic properties.

\subsection{Definition and basic API}

\begin{definition}[Edges induced on a vertex set]
  \label{def:edgesIn}
  \lean{SimpleGraph.edgesIn}
  \leanok
  Define
  \[
    \edgesIn{S} \;:=\; \{\,e \in E(G) : e \subseteq S\,\}.
  \]
\end{definition}

\begin{lemma}[Monotonicity]
  \label{lem:edgesIn-mono}
  \lean{SimpleGraph.edgesIn_mono}
  \leanok
  \uses{def:edgesIn}
  If $S \subseteq T$ then $\edgesIn{S} \subseteq \edgesIn{T}$.
\end{lemma}
\begin{proof}
  \leanok
  An edge $e \subseteq S$ also satisfies $e \subseteq T$.
\end{proof}

\begin{lemma}[Corner cases]
  \label{lem:edgesIn-corners}
  \lean{SimpleGraph.edgesIn_univ, SimpleGraph.edgesIn_empty, SimpleGraph.edgesIn_singleton, SimpleGraph.edgesIn_bot}
  \leanok
  \uses{def:edgesIn}
  $\edgesIn{V} = E(G)$, $\edgesIn{\emptyset} = \emptyset$, and
  $\edgesIn{\{v\}} = \emptyset$ for every $v \in V$. Moreover, the empty
  graph has no edges in any vertex set: $\edgesIn[\bot]{S} = \emptyset$.
\end{lemma}
\begin{proof}
  \leanok
  The first two are immediate from the definition. The singleton case
  uses that $G$ is a simple graph, hence loopless: no edge has both
  endpoints equal to a fixed $v$.
\end{proof}

\subsection{Vertex deletion}

The next two lemmas describe what happens to $\edgesIn{S}$ when $S$ is
obtained by removing a single vertex $v$. Together they yield the
edge-count identity that feeds the minimum-degree argument for Laman
graphs in subsequent chapters.

Write $I_G(v) := \{\,e \in E(G) : v \in e\,\}$ for the set of edges of
$G$ incident to $v$ (mathlib's \texttt{SimpleGraph.incidenceSet}).

\begin{lemma}[Edges missing a vertex]
  \label{lem:mem-edgesIn-compl-singleton}
  \lean{SimpleGraph.mem_edgesIn_compl_singleton, SimpleGraph.edgesIn_compl_singleton}
  \leanok
  \uses{def:edgesIn}
  For every $v \in V$,
  \[
    \edgesIn{\{v\}^{c}} \;=\; E(G) \,\setminus\, I_G(v).
  \]
  Equivalently, an edge $e$ avoids $v$ iff $e \in E(G)$ and $v \notin e$.
\end{lemma}
\begin{proof}
  \leanok
  An edge has both endpoints in $\{v\}^c$ exactly when neither
  endpoint equals $v$, i.e.\ when $v \notin e$. The set form follows by
  taking the complement of the incidence set in $E(G)$.
\end{proof}

\begin{lemma}[Vertex-deletion partition of edges]
  \label{lem:ncard-edgesIn-compl-singleton}
  \lean{SimpleGraph.ncard_edgesIn_compl_singleton_add_ncard_incidenceSet}
  \leanok
  \uses{lem:mem-edgesIn-compl-singleton}
  If $V$ is finite then, for every $v \in V$,
  \[
    \bigl|\edgesIn{\{v\}^{c}}\bigr| \;+\; |I_G(v)| \;=\; |E(G)|.
  \]
\end{lemma}
\begin{proof}
  \leanok
  By \cref{lem:mem-edgesIn-compl-singleton} the edge set partitions
  as $\edgesIn{\{v\}^c} \sqcup I_G(v)$, so the cardinalities add.
\end{proof}

\subsection{Modularity: intersection, union, and supermodular counts}

The selector $\edgesIn{\cdot}$ distributes exactly over intersection,
but only sub-distributes over union: a crossing edge with one endpoint
in $S \setminus T$ and the other in $T \setminus S$ lies in
$\edgesIn{S \cup T}$ but in neither $\edgesIn{S}$ nor $\edgesIn{T}$. The
cardinality consequence is the supermodular bound below, which feeds
the tight-subset lattice in \cref{thm:isTightOn-union-inter}.

\begin{lemma}[Distribution over intersection]
  \label{lem:edgesIn-inter}
  \lean{SimpleGraph.edgesIn_inter}
  \leanok
  \uses{def:edgesIn}
  $\edgesIn{S \cap T} = \edgesIn{S} \cap \edgesIn{T}$.
\end{lemma}
\begin{proof}
  \leanok
  An edge has both endpoints in $S \cap T$ iff both endpoints lie in
  $S$ \emph{and} both endpoints lie in $T$.
\end{proof}

\begin{lemma}[Supermodularity of edge counts]
  \label{lem:edgesIn-ncard-add-le}
  \lean{SimpleGraph.edgesIn_ncard_add_le}
  \leanok
  \uses{lem:edgesIn-mono, lem:edgesIn-inter}
  If $V$ is finite, then for all $S, T \subseteq V$,
  \[
    \bigl|\edgesIn{S}\bigr| + \bigl|\edgesIn{T}\bigr|
    \;\le\;
    \bigl|\edgesIn{S \cup T}\bigr| + \bigl|\edgesIn{S \cap T}\bigr|.
  \]
\end{lemma}
\begin{proof}
  \leanok
  By monotonicity (\cref{lem:edgesIn-mono}) we have
  $\edgesIn{S} \cup \edgesIn{T} \subseteq \edgesIn{S \cup T}$, while
  $\edgesIn{S} \cap \edgesIn{T} = \edgesIn{S \cap T}$ by
  \cref{lem:edgesIn-inter}. Substituting these into the inclusion--%
  exclusion identity
  $|A| + |B| = |A \cup B| + |A \cap B|$ for $A = \edgesIn{S}$ and
  $B = \edgesIn{T}$ gives
  $|\edgesIn{S}| + |\edgesIn{T}| = |\edgesIn{S} \cup \edgesIn{T}|
    + |\edgesIn{S \cap T}|
    \le |\edgesIn{S \cup T}| + |\edgesIn{S \cap T}|$.
\end{proof}

\section{\texorpdfstring{$(k,\ell)$}{(k,l)}-sparsity and \texorpdfstring{$(k,\ell)$}{(k,l)}-tightness}
\label{sec:sparsity}

We now state the sparsity condition and develop its API. Throughout
this section, $k$ and $\ell$ are non-negative integers (think
$k = 2$, $\ell = 3$ for the Laman case in \cref{sec:intro}).

\subsection{Definitions}

\begin{definition}[$(k,\ell)$-sparse]
  \label{def:isSparse}
  \lean{SimpleGraph.IsSparse}
  \leanok
  \uses{def:edgesIn}
  $G$ is \emph{$(k,\ell)$-sparse} if for every finite $s \subseteq V$
  with $\ell \le k\,|s|$ one has
  \[
    \bigl|\edgesIn{s}\bigr| + \ell \;\le\; k\,|s|.
  \]
  The hypothesis $\ell \le k\,|s|$ ensures the bound is non-negative;
  phrasing the conclusion additively keeps the formalization free of
  $\mathbb{N}$-subtraction.
\end{definition}

\begin{definition}[$(k,\ell)$-tight]
  \label{def:isTight}
  \lean{SimpleGraph.IsTight}
  \leanok
  \uses{def:isSparse}
  $G$ is \emph{$(k,\ell)$-tight} if it is $(k,\ell)$-sparse and, on the
  full finite vertex type, the global bound is met with equality:
  \[
    |E(G)| + \ell \;=\; k\,|V|.
  \]
\end{definition}

\begin{definition}[Tight subset]
  \label{def:isTightOn}
  \lean{SimpleGraph.IsTightOn}
  \leanok
  \uses{def:edgesIn}
  A finite set $s \subseteq V$ is \emph{$(k,\ell)$-tight in $G$} if it
  attains the sparsity bound:
  \[
    \bigl|\edgesIn{s}\bigr| + \ell \;=\; k\,|s|.
  \]
  This is the local counterpart of \cref{def:isTight} and underlies
  the tight-subset lattice (\cref{thm:isTightOn-union-inter}).

  \emph{Terminology.} In a $(2, 3)$-sparse graph $G$, a $(2, 3)$-tight
  subset is what Jord\'an~\cite{jordan2016}~\S 2.1 calls a
  \emph{critical set} of $G$. The two terms are interchangeable; we
  use ``tight'' throughout to match the predicate name
  \texttt{IsTightOn}.
\end{definition}

\subsection{Basic API and the global edge bound}

\begin{lemma}[Subgraphs of sparse graphs are sparse]
  \label{lem:isSparse-mono-left}
  \lean{SimpleGraph.IsSparse.mono_left}
  \leanok
  \uses{def:isSparse}
  If $G \le G'$ and $G'$ is $(k,\ell)$-sparse, so is $G$.
\end{lemma}
\begin{proof}
  \leanok
  For any vertex set $s$, every edge of $G$ supported on $s$ is also
  an edge of $G'$ supported on $s$, so
  $|\edgesIn[G]{s}| \le |\edgesIn[G']{s}|$; the sparsity bound for
  $G'$ then bounds the count for $G$.
\end{proof}

\begin{lemma}[Global edge bound]
  \label{lem:isSparse-edgeSet-bound}
  \lean{SimpleGraph.IsSparse.edgeSet_ncard_add_le}
  \leanok
  \uses{def:isSparse}
  If $V$ is finite, $G$ is $(k,\ell)$-sparse, and $\ell \le k\,|V|$,
  then
  \[
    |E(G)| + \ell \;\le\; k\,|V|.
  \]
\end{lemma}
\begin{proof}
  \leanok
  Apply the sparsity bound at $s = V$ and use $\edgesIn{V} = E(G)$
  (\cref{lem:edgesIn-corners}).
\end{proof}

\begin{lemma}[A proper extension of a tight graph is not sparse]
  \label{lem:isTight-not-isSparse-of-lt}
  \lean{SimpleGraph.IsTight.not_isSparse_of_lt}
  \leanok
  \uses{def:isTight, lem:isSparse-edgeSet-bound}
  If $V$ is finite, $G$ is $(k,\ell)$-tight, and $G < H$, then $H$ is
  not $(k,\ell)$-sparse.
\end{lemma}
\begin{proof}
  \leanok
  By tightness $|E(G)| + \ell = k\,|V|$. Since $G < H$ we have
  $|E(G)| < |E(H)|$, and so $|E(H)| + \ell > k\,|V|$. By
  \cref{lem:isSparse-edgeSet-bound} this rules out
  $(k,\ell)$-sparsity of $H$.
\end{proof}

\subsection{Low-degree vertex and non-adjacency consequences}

Two specializations of the global edge bound at $(k, \ell) = (2, 3)$
feed the Henneberg constructions for both Laman graphs (Phase~5
milestone~1) and arbitrary $(2,3)$-sparse graphs (Phase~7 sparse-graph
reverse decomposition). The first uses the handshake identity to show
that some vertex has degree $\le 3$; the second applies the sparsity
bound at a 4-vertex subset to show that any three neighbors of a vertex
contain a non-adjacent pair.

\begin{lemma}[A vertex of degree at most 3]
  \label{lem:isSparse-exists-degree-le-three}
  \lean{SimpleGraph.IsSparse.exists_degree_le_three}
  \leanok
  \uses{def:isSparse, lem:isSparse-edgeSet-bound}
  Let $V$ be finite with $|V| \ge 2$ and let $G$ be $(2,3)$-sparse. Then
  there exists $v \in V$ with $G$-degree at most $3$.
\end{lemma}
\begin{proof}
  \leanok
  Suppose every vertex had degree at least $4$. Then the handshake
  identity gives $2|E(G)| = \sum_v \deg_G(v) \ge 4|V|$. But the
  global edge bound (\cref{lem:isSparse-edgeSet-bound} at
  $(k, \ell) = (2, 3)$, valid because $|V| \ge 2$) rewrites the
  left-hand side as $2|E(G)| \le 4|V| - 6 < 4|V|$, a contradiction.
\end{proof}

\begin{lemma}[Non-adjacent pair among three neighbors]
  \label{lem:isSparse-exists-nonadj-among-three-neighbors}
  \lean{SimpleGraph.IsSparse.exists_nonadj_among_three_neighbors}
  \leanok
  \uses{def:isSparse}
  Let $V$ be finite and let $G$ be $(2,3)$-sparse. Suppose $v \in V$ has
  three pairwise distinct neighbors $a, b, c$. Then at least one of the
  pairs $\{a, b\}$, $\{a, c\}$, $\{b, c\}$ is non-adjacent in $G$.
\end{lemma}
\begin{proof}
  \leanok
  Suppose all three pairs were adjacent. Then the 4-set $\{v, a, b, c\}$
  would span the 6 edges
  $\{v\text{-}a, v\text{-}b, v\text{-}c, a\text{-}b, a\text{-}c, b\text{-}c\}$,
  so $|\edgesIn{\{v, a, b, c\}}| \ge 6$. But $(2,3)$-sparsity at the
  4-set gives $|\edgesIn{\{v, a, b, c\}}| \le 2 \cdot 4 - 3 = 5$.
\end{proof}

\subsection{Transport along graph isomorphism}

Both sparsity and tightness are isomorphism invariants. The transport
is mediated by \texttt{Sym2.map} on edges and by the underlying
\texttt{Equiv} on vertices.

\begin{lemma}[$\edgesIn{\cdot}$ transports along isomorphisms]
  \label{lem:iso-image-edgesIn}
  \lean{SimpleGraph.Iso.image_edgesIn}
  \leanok
  \uses{def:edgesIn}
  Let $\varphi : G \simeq H$ be a graph isomorphism and $s \subseteq V$.
  Then
  $\operatorname{Sym2.map} \varphi \,[\, \edgesIn[G]{s}\,]
   = \edgesIn[H]{\varphi(s)}$.
\end{lemma}
\begin{proof}
  \leanok
  An unordered pair $\{u, v\}$ in the image set comes from a pair
  $\{p, q\} \subseteq s$ in $\edgesIn[G]{s}$ with $u = \varphi(p)$ and
  $v = \varphi(q)$, by case analysis on the symmetric pair's two
  representatives; $\varphi$ preserves adjacency in each direction.
\end{proof}

\begin{lemma}[Sparsity is preserved by isomorphism]
  \label{lem:isSparse-iso}
  \lean{SimpleGraph.IsSparse.iso}
  \leanok
  \uses{def:isSparse, lem:iso-image-edgesIn}
  If $\varphi : G \simeq H$ and $G$ is $(k,\ell)$-sparse, then so is $H$.
\end{lemma}
\begin{proof}
  \leanok
  For $s \subseteq W$ a finite set on the codomain, transport along
  $\varphi^{-1}$ to a set $s' \subseteq V$ of the same cardinality.
  By \cref{lem:iso-image-edgesIn},
  $|\edgesIn[H]{s}| = |\edgesIn[G]{s'}|$. Apply $(k,\ell)$-sparsity of
  $G$ at $s'$.
\end{proof}

\begin{lemma}[Tightness is preserved by isomorphism]
  \label{lem:isTight-iso}
  \lean{SimpleGraph.IsTight.iso}
  \leanok
  \uses{def:isTight, lem:isSparse-iso}
  If $\varphi : G \simeq H$ and $G$ is $(k,\ell)$-tight, then so is $H$.
\end{lemma}
\begin{proof}
  \leanok
  Sparsity transports by \cref{lem:isSparse-iso}. For the global edge
  identity, $\varphi$ induces bijections $E(G) \cong E(H)$ and
  $V \cong W$ (the latter via \texttt{Equiv}), so
  $|E(H)| = |E(G)|$ and $|W| = |V|$. Substituting these into the
  tightness equation for $G$ yields the one for $H$.
\end{proof}

\subsection{The tight-subset lattice}

The squeeze lemma below is a converse to the sparsity inequality
within a single sparse graph: a vertex set that hits the bound from
below must hit it on the nose, hence is tight. It is the
edge-counting input used in
\cref{lem:isSparse-typeII-reverse-blocker} to extract a tight set from
a sparsity violation on a candidate graph.

\begin{lemma}[Squeeze]
  \label{lem:isSparse-isTightOn-of-le}
  \lean{SimpleGraph.IsSparse.isTightOn_of_le}
  \leanok
  \uses{def:isSparse, def:isTightOn}
  Let $G$ be $(k,\ell)$-sparse, $s \subseteq V$ finite with
  $\ell \le k\,|s|$, and suppose
  \[
    k\,|s| \;\le\; \bigl|\edgesIn{s}\bigr| + \ell.
  \]
  Then $s$ is $(k,\ell)$-tight in $G$.
\end{lemma}
\begin{proof}
  \leanok
  The sparsity bound at $s$ gives the reverse inequality
  $|\edgesIn{s}| + \ell \le k\,|s|$. The two together force equality.
\end{proof}

The main structural fact of this section is that, inside a fixed
$(k,\ell)$-sparse graph, the family of tight subsets is closed under
unions and intersections, modulo a standard size proviso on the
intersection.

\begin{theorem}[Tight subsets are closed under union and intersection]
  \label{thm:isTightOn-union-inter}
  \lean{SimpleGraph.IsTightOn.union_inter}
  \leanok
  \uses{def:isSparse, def:isTightOn, lem:edgesIn-ncard-add-le}
  Let $V$ be finite and $G$ be $(k,\ell)$-sparse. Let $s, t \subseteq V$
  be finite vertex sets, both $(k,\ell)$-tight in $G$, and suppose
  $\ell \le k\,|s \cap t|$. Then both $s \cup t$ and $s \cap t$ are
  $(k,\ell)$-tight in $G$.
\end{theorem}
\begin{proof}
  \leanok
  Sparsity applied to $s \cap t$ (using the size hypothesis) and to
  $s \cup t$ (whose cardinality is at least $|s \cap t|$) gives
  \begin{align*}
    \bigl|\edgesIn{s \cap t}\bigr| + \ell &\le k\,|s \cap t|, \\
    \bigl|\edgesIn{s \cup t}\bigr| + \ell &\le k\,|s \cup t|.
  \end{align*}
  Adding these and using
  $|s \cup t| + |s \cap t| = |s| + |t|$
  yields
  \[
    \bigl|\edgesIn{s \cup t}\bigr|
      + \bigl|\edgesIn{s \cap t}\bigr|
      + 2\ell
    \;\le\; k\,|s| + k\,|t|.
  \]
  On the other hand, supermodularity
  (\cref{lem:edgesIn-ncard-add-le}) gives
  \[
    \bigl|\edgesIn{s}\bigr| + \bigl|\edgesIn{t}\bigr|
    \;\le\;
    \bigl|\edgesIn{s \cup t}\bigr|
      + \bigl|\edgesIn{s \cap t}\bigr|,
  \]
  and tightness of $s$ and $t$ rewrites the left side as
  $k\,|s| + k\,|t| - 2\ell$. Chaining the two inequalities forces
  every inequality to be an equality, i.e.\ both $s \cup t$ and
  $s \cap t$ are tight.
\end{proof}

Two refinements of \cref{thm:isTightOn-union-inter}'s union half drop
the size proviso on $s \cap t$ in exchange for an explicit set $F$ of
extra edges that closes the accounting. They specialize to the union
half of \cref{thm:isTightOn-union-inter} at $F = \emptyset$.

\begin{theorem}[Tight union with bonus edges]
  \label{thm:isTightOn-union-with-bonus}
  \lean{SimpleGraph.IsTightOn.union_with_bonus}
  \leanok
  \uses{def:isSparse, def:isTightOn, lem:edgesIn-ncard-add-le}
  Let $V$ be finite and $G$ be $(k,\ell)$-sparse. Let
  $s_1, s_2 \subseteq V$ be both $(k,\ell)$-tight, and let
  $F \subseteq \edgesIn{s_1 \cup s_2}$ be a finite set of edges
  disjoint from $\edgesIn{s_1} \cup \edgesIn{s_2}$, satisfying
  \[
    |\edgesIn{s_1 \cap s_2}| + \ell \;\le\; |F| + k\,|s_1 \cap s_2|.
  \]
  Then $s_1 \cup s_2$ is $(k,\ell)$-tight.
\end{theorem}
\begin{proof}
  \leanok
  Refined supermodularity packages the disjoint inclusion
  $\edgesIn{s_1} \cup \edgesIn{s_2} \cup F \subseteq \edgesIn{s_1 \cup s_2}$
  with $\edgesIn{s_1} \cap \edgesIn{s_2} = \edgesIn{s_1 \cap s_2}$
  to give
  \[
    |\edgesIn{s_1}| + |\edgesIn{s_2}| + |F|
    \;\le\;
    |\edgesIn{s_1 \cup s_2}| + |\edgesIn{s_1 \cap s_2}|.
  \]
  Tightness of $s_1$ and $s_2$ rewrites the left side as
  $k\,|s_1| + k\,|s_2| - 2\ell + |F|$. Combined with the
  $|s_1| + |s_2| = |s_1 \cup s_2| + |s_1 \cap s_2|$ identity and the
  hypothesis on $F$, this lower-bounds $|\edgesIn{s_1 \cup s_2}|$ by
  $k\,|s_1 \cup s_2| - \ell$, which against sparsity forces equality.
\end{proof}

\begin{theorem}[Tight extension by a vertex with $k$ boundary edges]
  \label{thm:isTightOn-insert-vertex-with-edges}
  \lean{SimpleGraph.IsTightOn.insert_vertex_with_edges}
  \leanok
  \uses{def:isSparse, def:isTightOn}
  Let $V$ be finite and $G$ be $(k,\ell)$-sparse. Let
  $s \subseteq V$ be $(k,\ell)$-tight, $w \notin s$, and let
  $F \subseteq \edgesIn{s \cup \{w\}}$ be a finite set of edges
  disjoint from $\edgesIn{s}$ with $|F| \ge k$. Then $s \cup \{w\}$
  is $(k,\ell)$-tight.
\end{theorem}
\begin{proof}
  \leanok
  The disjoint inclusion
  $\edgesIn{s} \cup F \subseteq \edgesIn{s \cup \{w\}}$ gives
  $|\edgesIn{s \cup \{w\}}| \ge |\edgesIn{s}| + |F|$. Tightness of $s$
  and $|F| \ge k$ rewrite this as
  $|\edgesIn{s \cup \{w\}}| \ge k\,|s| - \ell + k = k\,|s \cup \{w\}| - \ell$,
  which against sparsity at $s \cup \{w\}$ forces equality.
\end{proof}

\subsection{Per-pair tight-blocker witness (typeII reverse, sparse form)}
\label{sec:isSparse-typeII-reverse-blocker}

The combinatorial heart of the typeII Henneberg reverse: a sparsity
violation on the bridge-augmented candidate graph
$G' := G[\{w \ne v\}] \cup \{\{x, y\}\}$ extracts a $(2, 3)$-tight
blocker in $G$ containing both $x$ and $y$. This is the sparse-graph
fact behind both the Laman strengthened decomposition
(\cref{thm:isLaman-typeII-reverse-blocker}) and the Phase~7
sparse-graph reverse decomposition
(\cref{thm:isSparse-exists-typeI-or-typeII-reverse}). The Laman shell
adds only the iso~+~edge-count conversion
$\neg G'.\mathrm{IsLaman} \to \neg G'.\mathrm{IsSparse}$ on top.

\begin{lemma}[Per-pair tight blocker, sparse form]
  \label{lem:isSparse-typeII-reverse-blocker}
  \lean{SimpleGraph.IsSparse.typeII_reverse_blocker}
  \leanok
  \uses{def:isSparse, def:isTightOn,
    lem:isSparse-isTightOn-of-le}
  Let $V$ be finite, let $G$ be $(2, 3)$-sparse on $V$, and fix
  $v \in V$ and $x, y \in V$ both $\ne v$. Suppose the bridge-augmented
  candidate $G' := G[\{w \ne v\}] \cup \{\{x, y\}\}$ is \emph{not}
  $(2, 3)$-sparse. Then there exists a finite $S \subseteq V$ with
  $v \notin S$, $x, y \in S$, and $S$ $(2, 3)$-tight in $G$.
\end{lemma}
\begin{proof}
  \leanok
  Extract a violating finite $s' \subseteq \{w \ne v\}$ from
  $\neg G'.\mathrm{IsSparse}$. Lift to
  $S := \{w.\mathrm{val} : w \in s'\} \subseteq V \setminus \{v\}$.
  The edges of $G'$ on $s'$ correspond to edges of $G$ on $S$, plus
  possibly the bridge $\{x, y\}$ when $x, y \in s'$. (In the
  formalization, this ``correspond to'' is a named project-internal
  helper that gives the equality
  $\mathrm{Sym2.map}\;f \,[(G.\mathrm{comap}\;f).\edgesIn{s'}]
   = \edgesIn[G]{f[s']}$
  for $f = \mathrm{Subtype.val}$; the cardinality version it feeds
  needs injectivity of $\mathrm{Sym2.map}\;f$, which is itself a small
  lift of injectivity of $f$.) Case-split on $x, y \in s'$: in the
  bridge-included case, the $\le 1$ overcount yields a squeeze that
  forces $S$ tight in $G$ (\cref{lem:isSparse-isTightOn-of-le}); in
  the bridge-excluded case, the bridge contributes nothing and the
  count violates $G$-sparsity directly, contradicting the hypothesis
  (a vacuous branch).
\end{proof}
