\section{Frameworks and infinitesimal rigidity}
\label{sec:frameworks}

A $d$-dimensional \emph{framework} on a vertex set $V$ is a placement
$p : V \to \R^{d}$, where $\R^{d}$ denotes Euclidean $d$-space with its
standard inner product. The linear-algebraic formulation of
infinitesimal rigidity developed below --- rigidity map, infinitesimal
motions, trivial Euclidean motions, and infinitesimal rigidity as a
kernel-dimension bound --- is due to Asimow and Roth~\cite{asimowRoth1978}.
Given a graph $G$ on $V$, the \emph{rigidity map} at $p$ is the
$\R$-linear map sending an infinitesimal motion
$p' \in \R^{d \cdot |V|}$ to the family of edge-length derivatives
$e \mapsto \langle p_u - p_v,\, p'_u - p'_v \rangle$ indexed by the
edges $e = \{u, v\} \in E(G)$. Geometrically this is the differential
at $p$ of the squared-edge-length map, up to a factor of $2$.

A framework is \emph{infinitesimally rigid} when the kernel of the
rigidity map has dimension at most $d(d+1)/2$ --- the dimension of the
space of trivial Euclidean motions (translations and infinitesimal
rotations). The graph $G$ is \emph{generically rigid} in dimension $d$
when \emph{some} placement is infinitesimally rigid. The section closes
with the edge-count bound: every generically rigid graph in dimension
$d$ on $n$ vertices satisfies $d \cdot n \le |E(G)| + d(d+1)/2$;
specializing at $d = 2$ recovers the necessary half of Laman's theorem.

All definitions are stated for a general dimension $d \in \N$; the
Laman-relevant specialization to $d = 2$ is developed in
\cref{sec:laman-theorem}.

\subsection{Definitions}

\begin{definition}[Framework]
  \label{def:framework}
  \lean{SimpleGraph.Framework}
  \leanok
  A $d$-dimensional \emph{framework} on $V$ is a function
  $p : V \to \R^{d}$.
\end{definition}

\begin{lemma}[Dimension of the framework space]
  \label{lem:framework-finrank}
  \lean{SimpleGraph.Framework.finrank}
  \leanok
  \uses{def:framework}
  If $V$ is finite, then
  $\dim_{\R}\,(\mathrm{Framework}\,V\,d) = |V| \cdot d$.
\end{lemma}
\begin{proof}
  \leanok
  The framework space is a product of $|V|$ copies of $\R^{d}$, each of
  dimension $d$, so its dimension is $|V| \cdot d$
  ($\mathrm{Module.finrank\_pi\_fintype}$).
\end{proof}

\begin{definition}[Rigidity map]
  \label{def:rigidityMap}
  \lean{SimpleGraph.RigidityMap}
  \leanok
  \uses{def:framework}
  For a graph $G$ on $V$ and a framework $p : V \to \R^{d}$, the
  \emph{rigidity map} at $p$ is the $\R$-linear map
  \[
    \mathrm{RigidityMap}\,G\,p :
    \mathrm{Framework}\,V\,d \;\to_{\R}\; \bigl(E(G) \to \R\bigr)
  \]
  with
  \[
    (\mathrm{RigidityMap}\,G\,p)(p')(\{u, v\})
    \;=\;
    \langle p_u - p_v,\; p'_u - p'_v \rangle.
  \]
\end{definition}
The entry formula is symmetric under interchange of $u$ and $v$ (each
factor changes sign), so it descends from the unordered pair
$\{u, v\}$ to a single scalar, and it is $\R$-linear in $p'$.
\begin{fmlnote}
  The rigidity map is defined directly by the entry formula above,
  rather than as the differential of the squared-edge-length map, so
  the rank and kernel arguments below require no differentiability
  machinery.
\end{fmlnote}

\begin{lemma}[Rigidity-map entry formula]
  \label{lem:rigidityMap-apply}
  \lean{SimpleGraph.rigidityMap_apply}
  \leanok
  \uses{def:rigidityMap}
  For $u, v \in V$ with $\{u, v\} \in E(G)$ and $p, p' : V \to \R^{d}$,
  \[
    (\mathrm{RigidityMap}\,G\,p)(p')(\{u, v\})
    \;=\;
    \langle p_u - p_v,\; p'_u - p'_v \rangle.
  \]
\end{lemma}
\begin{proof}
  \leanok
  Immediate from the definition.
\end{proof}

\begin{definition}[Infinitesimally rigid framework]
  \label{def:isInfinitesimallyRigid}
  \lean{SimpleGraph.IsInfinitesimallyRigid}
  \leanok
  \uses{def:rigidityMap}
  For finite $V$, a framework $p$ is \emph{infinitesimally rigid} for
  $G$ if
  \[
    \dim_{\R}\,\bigl(\ker (\mathrm{RigidityMap}\,G\,p)\bigr)
    \;\le\;
    \frac{d(d+1)}{2}.
  \]
\end{definition}

\begin{definition}[Generically rigid graph]
  \label{def:isGenericallyRigid}
  \lean{SimpleGraph.IsGenericallyRigid}
  \leanok
  \uses{def:isInfinitesimallyRigid}
  A graph $G$ on finite $V$ is \emph{generically rigid} in dimension
  $d$ if some framework $p : V \to \R^{d}$ is infinitesimally rigid
  for $G$.
\end{definition}

\begin{definition}[Generically rigid graph, injective form]
  \label{def:isGenericallyRigidInj}
  \lean{SimpleGraph.IsGenericallyRigidInj}
  \leanok
  \uses{def:isInfinitesimallyRigid}
  A graph $G$ on finite $V$ is \emph{generically rigid with an
  injective placement} in dimension $d$ if some framework
  $p : V \to \R^{d}$ is both infinitesimally rigid for $G$ and
  injective.
\end{definition}
The injective form is strictly stronger than plain generic rigidity
(\cref{def:isGenericallyRigid}). The Henneberg induction of
\cref{sec:laman-theorem} maintains an injective placement at each step,
since the per-move preservation arguments require the two attaching
vertices to be placed at distinct points, and weakens to plain generic
rigidity only at the end.

\subsection{Rank and kernel bounds}

Two estimates on the rigidity map are used in the edge-count bound
below: its kernel only grows as edges are removed, and its rank is
bounded by the edge count.

\begin{lemma}[Kernel monotonicity]
  \label{lem:rigidityMap-ker-mono}
  \lean{SimpleGraph.rigidityMap_ker_mono}
  \leanok
  \uses{def:rigidityMap}
  If $G \le G'$ then for every framework $p$,
  \[
    \ker (\mathrm{RigidityMap}\,G'\,p)
    \;\le\;
    \ker (\mathrm{RigidityMap}\,G\,p).
  \]
\end{lemma}
\begin{proof}
  \leanok
  The edges of $G$ are among those of $G'$. An infinitesimal motion in
  $\ker (\mathrm{RigidityMap}\,G'\,p)$ has rigidity-map value $0$ on
  every edge of $G'$, hence on every edge of $G$, so it lies in
  $\ker (\mathrm{RigidityMap}\,G\,p)$.
\end{proof}

\begin{lemma}[Rank is bounded by edge count]
  \label{lem:rigidityMap-finrank-range-le}
  \lean{SimpleGraph.rigidityMap_finrank_range_le}
  \leanok
  \uses{def:rigidityMap}
  If $V$ is finite, then for every framework $p$,
  \[
    \dim_{\R}\,\bigl(\mathrm{range}\,(\mathrm{RigidityMap}\,G\,p)\bigr)
    \;\le\;
    |E(G)|.
  \]
\end{lemma}
\begin{proof}
  \leanok
  The range is a submodule of $E(G) \to \R$, whose dimension is
  $|E(G)|$ by $\mathrm{Module.finrank\_pi}$.
\end{proof}

\subsection{Graph monotonicity}

Generic rigidity is a monotone property in the graph: adding edges
preserves rigidity at any fixed placement.

\begin{lemma}[Infinitesimal rigidity is monotone]
  \label{lem:isInfinitesimallyRigid-mono}
  \lean{SimpleGraph.IsInfinitesimallyRigid.mono}
  \leanok
  \uses{def:isInfinitesimallyRigid, lem:rigidityMap-ker-mono}
  If $G \le G'$ and $p$ is infinitesimally rigid for $G$, then $p$ is
  infinitesimally rigid for $G'$.
\end{lemma}
\begin{proof}
  \leanok
  By \cref{lem:rigidityMap-ker-mono} the kernel of
  $\mathrm{RigidityMap}\,G'\,p$ is contained in that of
  $\mathrm{RigidityMap}\,G\,p$, so its dimension is at most the
  latter's, which is at most $d(d+1)/2$ by hypothesis.
\end{proof}

\begin{lemma}[Generic rigidity is monotone]
  \label{lem:isGenericallyRigid-mono}
  \lean{SimpleGraph.IsGenericallyRigid.mono}
  \leanok
  \uses{def:isGenericallyRigid, lem:isInfinitesimallyRigid-mono}
  If $G \le G'$ and $G$ is generically rigid in dimension $d$, then so
  is $G'$.
\end{lemma}
\begin{proof}
  \leanok
  Apply \cref{lem:isInfinitesimallyRigid-mono} at the same placement
  witness.
\end{proof}

\begin{lemma}[Injective generic rigidity is monotone]
  \label{lem:isGenericallyRigidInj-mono}
  \lean{SimpleGraph.IsGenericallyRigidInj.mono}
  \leanok
  \uses{def:isGenericallyRigidInj, lem:isInfinitesimallyRigid-mono}
  If $G \le G'$ and $G$ is injectively generically rigid in dimension
  $d$, then so is $G'$.
\end{lemma}
\begin{proof}
  \leanok
  At the same placement: rigidity transports by
  \cref{lem:isInfinitesimallyRigid-mono}, and injectivity is a property
  of the placement alone.
\end{proof}

\subsection{Transport along graph isomorphism}
\label{sec:rigidity-iso-transport}

A graph isomorphism $\varphi : G \simeq H$ carries a rigid placement
of $G$ to a rigid placement of $H$ by precomposition with the
underlying vertex bijection. The Henneberg induction
(\cref{sec:laman-theorem}) uses this to work up to graph isomorphism
rather than on literal equalities of graphs.

\begin{lemma}[Infinitesimal rigidity transports along graph isomorphism]
  \label{lem:isInfinitesimallyRigid-iso}
  \lean{SimpleGraph.IsInfinitesimallyRigid.iso}
  \leanok
  \uses{def:isInfinitesimallyRigid}
  Let $\varphi : G \simeq H$ be a graph isomorphism and $p$ a
  framework of $G$. If $p$ is infinitesimally rigid for $G$ then
  $p \circ \varphi^{-1}$ is infinitesimally rigid for $H$.
\end{lemma}
\begin{proof}
  \leanok
  Precomposition with $\varphi$ (resp.\ $\varphi^{-1}$) gives mutually
  inverse $\R$-linear maps between the two kernels: on edges, the
  rigidity-map entry at $\{u, v\}$ for $H$ equals the one at
  $\{\varphi^{-1} u, \varphi^{-1} v\}$ for $G$ (the inner-product
  formula is the same, and $\varphi$ preserves adjacency). The
  resulting $\R$-linear equivalence has the same finite-dimensional
  rank, so the kernel-dim bound for $G$ at $p$ transports to the
  one for $H$ at $p \circ \varphi^{-1}$.
\end{proof}

\begin{lemma}[Generic rigidity transports along graph isomorphism]
  \label{lem:isGenericallyRigid-iso}
  \lean{SimpleGraph.IsGenericallyRigid.iso}
  \leanok
  \uses{def:isGenericallyRigid, lem:isInfinitesimallyRigid-iso}
  If $\varphi : G \simeq H$ and $G$ is generically rigid in dimension
  $d$, then so is $H$.
\end{lemma}
\begin{proof}
  \leanok
  Transport the witness placement of $G$ via \cref{lem:isInfinitesimallyRigid-iso}.
\end{proof}

\begin{lemma}[Injective generic rigidity transports along graph isomorphism]
  \label{lem:isGenericallyRigidInj-iso}
  \lean{SimpleGraph.IsGenericallyRigidInj.iso}
  \leanok
  \uses{def:isGenericallyRigidInj, lem:isInfinitesimallyRigid-iso}
  If $\varphi : G \simeq H$ and $G$ is injectively generically rigid
  in dimension $d$, then so is $H$.
\end{lemma}
\begin{proof}
  \leanok
  Transport via \cref{lem:isInfinitesimallyRigid-iso}; the new
  placement $p \circ \varphi^{-1}$ is injective because $p$ is and
  $\varphi^{-1}$ is a bijection.
\end{proof}

\subsection{Openness of infinitesimal rigidity}
\label{sec:rigidity-openness}

Infinitesimal rigidity is an \emph{open} condition on the placement:
if $p_0$ is infinitesimally rigid then so is every placement in some
neighborhood of $p_0$. Perturbation therefore strengthens a rigid
placement to one also satisfying an additional open condition --- for
instance, a non-collinear triple --- as needed in
\cref{thm:typeII-isGenericallyRigidInj-two}.

\begin{lemma}[Infinitesimal rigidity is an open condition on the placement]
  \label{lem:isInfinitesimallyRigid-eventually}
  \lean{SimpleGraph.IsInfinitesimallyRigid.eventually}
  \leanok
  \uses{def:isInfinitesimallyRigid, lem:rigidityMap-apply}
  Let $V$ be finite. If $p_0 : V \to \R^{d}$ is infinitesimally rigid
  for $G$, then there is a neighborhood of $p_0$ in the framework
  space on which every placement is infinitesimally rigid for $G$.
\end{lemma}
\begin{proof}
  \leanok
  Set $r = \dim\,(\mathrm{range}\,(\mathrm{RigidityMap}\,G\,p_0))$ and
  pick a basis of the range; lift each basis vector through
  $\mathrm{RigidityMap}\,G\,p_0$ to preimages
  $\xi_1, \dots, \xi_r \in \mathrm{Framework}\,V\,d$. The map
  $p \mapsto \bigl(\mathrm{RigidityMap}\,G\,p\,(\xi_i)\bigr)_{i = 1}^{r}$
  is continuous in $p$ (each entry expands to an inner product, jointly
  continuous in $(p, \xi)$, with the $\xi_i$ fixed; cf.\
  \cref{lem:rigidityMap-apply}), and at $p_0$ the values
  $\mathrm{RigidityMap}\,G\,p_0\,(\xi_i)$ form a basis of the range,
  hence are linearly independent. By openness of linear independence
  in finite-dimensional normed spaces, this remains true on an open
  neighborhood of $p_0$, giving
  $r \le \dim\,(\mathrm{range}\,(\mathrm{RigidityMap}\,G\,p))$ for $p$
  in that neighborhood.

  Rank--nullity at both $p_0$ and $p$ now bounds the kernel dimension
  of $\mathrm{RigidityMap}\,G\,p$ by $d(d+1)/2$.
\end{proof}

\subsection{The edge-count bound}

The rank and kernel estimates above combine through rank--nullity into
the edge-count bound.

\begin{theorem}[Edge-count bound for generically rigid graphs]
  \label{thm:isGenericallyRigid-card-mul-le}
  \lean{SimpleGraph.IsGenericallyRigid.card_mul_le}
  \leanok
  \uses{def:isGenericallyRigid, lem:framework-finrank,
    lem:rigidityMap-finrank-range-le}
  Let $V$ be finite and $G$ generically rigid in dimension $d$. Then
  \[
    d \cdot |V| \;\le\; |E(G)| + \frac{d(d+1)}{2}.
  \]
  In particular, specializing at $d = 2$ gives
  $2 |V| \le |E(G)| + 3$, the necessary half of Laman's theorem.
\end{theorem}
\begin{proof}
  \leanok
  Pick an infinitesimally rigid framework $p$ witnessing generic
  rigidity. Rank--nullity for $\mathrm{RigidityMap}\,G\,p$ reads
  \[
    \dim\,\mathrm{Framework}\,V\,d
    \;=\;
    \dim\,(\mathrm{range}) + \dim\,(\ker).
  \]
  By \cref{lem:framework-finrank} the left-hand side equals $d \cdot |V|$;
  by \cref{lem:rigidityMap-finrank-range-le} the rank is at most
  $|E(G)|$; and by definition of infinitesimal rigidity the nullity
  is at most $d(d+1)/2$. Combining yields the claim.
\end{proof}
\begin{fmlnote}
  The bound is stated additively, $d \cdot |V| \le |E(G)| + d(d+1)/2$,
  rather than in the equivalent form
  $d \cdot |V| - d(d+1)/2 \le |E(G)|$, so that the Lean statement avoids
  truncated natural-number subtraction.
\end{fmlnote}

\subsection{Worked example: \texorpdfstring{$K_2$}{K2} is generically rigid in every dimension}

As a worked example, and as the base case for the Henneberg induction
of \cref{sec:laman-theorem}, the complete graph $K_2$ is generically
rigid --- indeed injectively so --- in every dimension.

\begin{theorem}[$K_2$ is injectively generically rigid in every positive dimension]
  \label{thm:top-fin-two-isGenericallyRigidInj}
  \lean{SimpleGraph.top_fin_two_isGenericallyRigidInj}
  \leanok
  \uses{def:isGenericallyRigidInj, lem:framework-finrank,
    lem:rigidityMap-finrank-range-le, lem:rigidityMap-apply}
  For every $d \ge 1$, the complete graph on two vertices is
  injectively generically rigid in dimension $d$.
\end{theorem}
\begin{proof}
  \leanok
  Take $p_0 = 0$ and $p_1 = e_0$ (the first standard basis vector);
  the placement is injective ($p_0 \ne p_1$). The single-edge image
  evaluates at the swap motion $p' = (e_0, 0)$ to
  $\langle 0 - e_0,\, e_0 - 0 \rangle = -1 \ne 0$
  (\cref{lem:rigidityMap-apply}), so the range has dimension at least
  $1$. The framework space has dimension $2d$
  (\cref{lem:framework-finrank}), so rank--nullity gives
  $\dim\,(\ker) \le 2d - 1$. For $d \ge 1$ the elementary inequality
  $2d - 1 \le d(d+1)/2$ (rewriting as $(d-1)(d-2) \ge 0$) closes the
  kernel-dimension bound to $d(d+1)/2$.
\end{proof}

\begin{theorem}[$K_2$ is generically rigid in every dimension]
  \label{thm:top-fin-two-isGenericallyRigid}
  \lean{SimpleGraph.top_fin_two_isGenericallyRigid}
  \leanok
  \uses{thm:top-fin-two-isGenericallyRigidInj}
  For every $d \in \N$, the complete graph on two vertices is
  generically rigid in dimension $d$.
\end{theorem}
\begin{proof}
  \leanok
  For $d \ge 1$, weaken \cref{thm:top-fin-two-isGenericallyRigidInj}
  by forgetting injectivity. For $d = 0$ the framework space itself
  is $0$-dimensional, so every placement has rigidity-map kernel of
  dimension $0$ and the bound $0 \le 0$ holds.
\end{proof}
